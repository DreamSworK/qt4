68 {
69  QGLGlyphTexture(const QGLContext *ctx)
70  : m_fbo(0)
71  , m_width(0)
72  , m_height(0)
73  {
74  if (ctx && QGLFramebufferObject::hasOpenGLFramebufferObjects() && !ctx->d_ptr->workaround_brokenFBOReadBack)
75  glGenFramebuffers(1, &m_fbo);
76 
77 #ifdef QT_GL_TEXTURE_GLYPH_CACHE_DEBUG
78  qDebug(" -> QGLGlyphTexture() %p for context %p.", this, ctx);
79 #endif
80  }
81 
82  ~QGLGlyphTexture() {
83  const QGLContext *ctx = QGLContext::currentContext();
84 #ifdef QT_GL_TEXTURE_GLYPH_CACHE_DEBUG
85  qDebug("~QGLGlyphTexture() %p for context %p.", this, ctx);
86 #endif
87  // At this point, the context group is made current, so it's safe to
88  // release resources without a makeCurrent() call
89  if (ctx) {
90  if (m_fbo)
91  glDeleteFramebuffers(1, &m_fbo);
92  if (m_width || m_height)
93  glDeleteTextures(1, &m_texture);
94  }
95  }
96 
97  GLuint m_texture;
98  GLuint m_fbo;
99  int m_width;
100  int m_height;
101 };
