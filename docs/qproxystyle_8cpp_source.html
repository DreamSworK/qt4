81 void QProxyStylePrivate::ensureBaseStyle() const
82 {
83  Q_Q(const QProxyStyle);
84 
85  if (baseStyle)
86  return;
87 
88  if (!baseStyle && !QApplicationPrivate::styleOverride.isEmpty()) {
89  baseStyle = QStyleFactory::create(QApplicationPrivate::styleOverride);
90  if (baseStyle) {
91  // If baseStyle is an instance of the same proxyStyle
92  // we destroy it and fall back to the desktop style
93  if (qstrcmp(baseStyle->metaObject()->className(),
94  q->metaObject()->className()) == 0) {
95  delete baseStyle;
96  baseStyle = 0;
97  }
98  }
99  }
100 
101  if (!baseStyle) // Use application desktop style
102  baseStyle = QStyleFactory::create(QApplicationPrivate::desktopStyleKey());
103 
104  if (!baseStyle) // Fallback to windows style
105  baseStyle = QStyleFactory::create(QLatin1String("windows"));
106 
107  baseStyle->setProxy(const_cast<QProxyStyle*>(q));
108  baseStyle->setParent(const_cast<QProxyStyle*>(q)); // Take ownership
109 }
