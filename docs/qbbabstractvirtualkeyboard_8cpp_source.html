68 void QBBAbstractVirtualKeyboard::setInputHintsFromWidget(QWidget *focusWidget)
69 {
70  if (focusWidget) {
71  const Qt::InputMethodHints hints = focusWidget->inputMethodHints();
72  if (hints & Qt::ImhEmailCharactersOnly)
73  setKeyboardMode(QBBAbstractVirtualKeyboard::Email);
74  else if (hints & Qt::ImhDialableCharactersOnly)
75  setKeyboardMode(QBBAbstractVirtualKeyboard::Phone);
76  else if (hints & Qt::ImhUrlCharactersOnly)
77  setKeyboardMode(QBBAbstractVirtualKeyboard::Web);
78  else if (hints & Qt::ImhFormattedNumbersOnly || hints & Qt::ImhDigitsOnly)
79  setKeyboardMode(QBBAbstractVirtualKeyboard::NumPunc);
80  else
81  setKeyboardMode(QBBAbstractVirtualKeyboard::Default);
82  } else {
83  setKeyboardMode(QBBAbstractVirtualKeyboard::Default);
