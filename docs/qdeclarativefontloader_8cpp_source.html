91 void QDeclarativeFontObject::download(const QUrl &url, QNetworkAccessManager *manager)
92 {
93  QNetworkRequest req(url);
94  req.setAttribute(QNetworkRequest::HttpPipeliningAllowedAttribute, true);
95  reply = manager->get(req);
96  QObject::connect(reply, SIGNAL(finished()), this, SLOT(replyFinished()));
97 }

99 void QDeclarativeFontObject::replyFinished()
100 {
101  if (reply) {
102  redirectCount++;
103  if (redirectCount < FONTLOADER_MAXIMUM_REDIRECT_RECURSION) {
104  QVariant redirect = reply->attribute(QNetworkRequest::RedirectionTargetAttribute);
105  if (redirect.isValid()) {
106  QUrl url = reply->url().resolved(redirect.toUrl());
107  QNetworkAccessManager *manager = reply->manager();
108  reply->deleteLater();
109  reply = 0;
110  download(url, manager);
111  return;
112  }
113  }
114  redirectCount = 0;
115 
116  if (!reply->error()) {
117  id = QFontDatabase::addApplicationFontFromData(reply->readAll());
118  if (id != -1)
119  emit fontDownloaded(QFontDatabase::applicationFontFamilies(id).at(0), QDeclarativeFontLoader::Ready);
120  else
121  emit fontDownloaded(QString(), QDeclarativeFontLoader::Error);
122  } else {
123  emit fontDownloaded(QString(), QDeclarativeFontLoader::Error);
124  }
125  reply->deleteLater();
126  reply = 0;
127  }
128 }

197 void QDeclarativeFontLoader::setSource(const QUrl &url)
198 {
199  Q_D(QDeclarativeFontLoader);
200  if (url == d->url)
201  return;
202  d->url = qmlContext(this)->resolvedUrl(url);
203  emit sourceChanged();
204 
205 #ifndef QT_NO_LOCALFILE_OPTIMIZED_QML
206  QString localFile = QDeclarativeEnginePrivate::urlToLocalFileOrQrc(d->url);
207  if (!localFile.isEmpty()) {
208  if (!d->fonts.contains(d->url)) {
209  int id = QFontDatabase::addApplicationFont(localFile);
210  if (id != -1) {
211  updateFontInfo(QFontDatabase::applicationFontFamilies(id).at(0), Ready);
212  QDeclarativeFontObject *fo = new QDeclarativeFontObject(id);
213  d->fonts[d->url] = fo;
214  } else {
215  updateFontInfo(QString(), Error);
216  }
217  } else {
218  updateFontInfo(QFontDatabase::applicationFontFamilies(d->fonts[d->url]->id).at(0), Ready);
219  }
220  } else
221 #endif
222  {
223  if (!d->fonts.contains(d->url)) {
224  QDeclarativeFontObject *fo = new QDeclarativeFontObject;
225  d->fonts[d->url] = fo;
226  fo->download(d->url, qmlEngine(this)->networkAccessManager());
227  d->status = Loading;
228  emit statusChanged();
229  QObject::connect(fo, SIGNAL(fontDownloaded(QString,QDeclarativeFontLoader::Status)),
230  this, SLOT(updateFontInfo(QString,QDeclarativeFontLoader::Status)));
231  } else {
232  QDeclarativeFontObject *fo = d->fonts[d->url];
233  if (fo->id == -1) {
234  d->status = Loading;
235  emit statusChanged();
236  QObject::connect(fo, SIGNAL(fontDownloaded(QString,QDeclarativeFontLoader::Status)),
237  this, SLOT(updateFontInfo(QString,QDeclarativeFontLoader::Status)));
238  }
239  else
240  updateFontInfo(QFontDatabase::applicationFontFamilies(fo->id).at(0), Ready);
241  }
242  }
243 }

245 void QDeclarativeFontLoader::updateFontInfo(const QString& name, QDeclarativeFontLoader::Status status)
246 {
247  Q_D(QDeclarativeFontLoader);
248 
249  if (name != d->name) {
