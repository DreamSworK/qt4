262 static int getDisplayId(const QString &spec)
263 {
264  QRegExp regexp(QLatin1String(":(\\d+)\\b"));
265  if (regexp.lastIndexIn(spec) != -1) {
266  const QString capture = regexp.cap(1);
267  return capture.toInt();
268  }
269  return 0;

288 {
289  if (!realScreen) {

292  }
293
