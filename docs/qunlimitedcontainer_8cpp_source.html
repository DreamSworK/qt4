63 {
64  const Expression::List::iterator end(m_operands.end());
65  Expression::List::iterator it(m_operands.begin());
66  int evaled = 0;
67 
68  for(; it != end; ++it)
69  {
70  Q_ASSERT((*it));
71  rewrite((*it), (*it)->compress(context), context);
72  if((*it)->isEvaluated())
73  ++evaled;
74  }
75 
76  return evaled == m_operands.count();
