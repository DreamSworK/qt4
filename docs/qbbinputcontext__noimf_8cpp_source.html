79 bool QBBInputContext::filterEvent( const QEvent *event )
80 {
81  if (hasPhysicalKeyboard())
82  return false;
83 
84  if (event->type() == QEvent::CloseSoftwareInputPanel) {
85  mVirtualKeyboard.hideKeyboard();
86 #if defined(QBBINPUTCONTEXT_DEBUG)
87  qDebug() << "QBB: hiding virtual keyboard";
88 #endif
89  return false;
90  }
91 
92  if (event->type() == QEvent::RequestSoftwareInputPanel) {
93  mVirtualKeyboard.showKeyboard();
94 #if defined(QBBINPUTCONTEXT_DEBUG)
95  qDebug() << "QBB: requesting virtual keyboard";
96 #endif
97  return false;
98  }
99 
100  return false;
101 
102 }

110 {

114  QInputContext::setFocusWidget(w);
115 
116  if (w) {
117  // Special case for item view which should not show the keyboard when focused
118  if (qobject_cast<QAbstractItemView*>(w))
119  return;
120 
121  mVirtualKeyboard.setInputHintsFromWidget(w);
122  mVirtualKeyboard.showKeyboard();
123  } else {
124  mVirtualKeyboard.hideKeyboard();
125  }
