76 void QBBScreenEventThread::run()
77 {
78  screen_event_t event;
79 
80  // create screen event
81  errno = 0;
82  int result = screen_create_event(&event);
83  if (result)
84  qFatal("QBB: failed to create screen event, errno=%d", errno);
85 
86 #if defined(QBBSCREENEVENTTHREAD_DEBUG)
87  qDebug() << "QBB: screen event thread started";
88 #endif
89 
90  // loop indefinitely
91  while (!mQuit) {
92 
93  // block until screen event is available
94  errno = 0;
95  result = screen_get_event(mContext, event, -1);
96  if (result)
97  qFatal("QBB: failed to get screen event, errno=%d", errno);
98 
99  // process received event
100  errno = 0;
101  int qnxType;
102  int result = screen_get_event_property_iv(event, SCREEN_PROPERTY_TYPE, &qnxType);
103  if (result)
104  qFatal("QBB: failed to query screen event type, errno=%d", errno);
105 
106  if (qnxType == SCREEN_EVENT_USER) {
107  // treat all screen user events as shutdown requests
108  #if defined(QBBSCREENEVENTTHREAD_DEBUG)
109  qDebug() << "QBB: QNX screen user event";
110  #endif
111  mQuit = true;
112  } else {
113  mEventHandler->handleEvent(event, qnxType);
114  }
115  }
116 
117 #if defined(QBBSCREENEVENTTHREAD_DEBUG)
118  qDebug() << "QBB: screen event thread stopped";
119 #endif
120 
121  // cleanup
122  screen_destroy_event(event);
123 }

125 void QBBScreenEventThread::shutdown()
126 {
127  screen_event_t event;
128 
129  // create screen event
130  errno = 0;
131  int result = screen_create_event(&event);
132  if (result)
133  qFatal("QBB: failed to create screen event, errno=%d", errno);
134 
135  // set the event type as user
136  errno = 0;
137  int type = SCREEN_EVENT_USER;
138  result = screen_set_event_property_iv(event, SCREEN_PROPERTY_TYPE, &type);
139  if (result)
140  qFatal("QBB: failed to set screen event type, errno=%d", errno);
141 
142  // NOTE: ignore SCREEN_PROPERTY_USER_DATA; treat all user events as shutdown events
143 
144  // post event to event loop so it will wake up and die
145  errno = 0;
146  result = screen_send_event(mContext, event, getpid());
147  if (result)
148  qFatal("QBB: failed to set screen event type, errno=%d", errno);
149 
150  // cleanup
151  screen_destroy_event(event);
152 
153 #if defined(QBBSCREENEVENTTHREAD_DEBUG)
154  qDebug() << "QBB: screen event thread shutdown begin";
155 #endif
156 
157  // block until thread terminates
158  wait();
159 
160 #if defined(QBBSCREENEVENTTHREAD_DEBUG)
161  qDebug() << "QBB: screen event thread shutdown end";
162 #endif
163 }
