517 bool QVideoFrame::map(QAbstractVideoBuffer::MapMode mode)
518 {
519  if (d->buffer != 0 && d->data == 0) {
520  Q_ASSERT(d->bytesPerLine == 0);
521  Q_ASSERT(d->mappedBytes == 0);
522 
523  d->data = d->buffer->map(mode, &d->mappedBytes, &d->bytesPerLine);
524 
525  return d->data != 0;
526  }
527 
528  return false;
529 }
