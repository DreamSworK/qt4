229 void QRubberBand::changeEvent(QEvent *e)
230 {
231  QWidget::changeEvent(e);
232  switch (e->type()) {
233  case QEvent::ParentChange:
234  if (parent()) {
235  setWindowFlags(windowFlags() & ~RUBBERBAND_WINDOW_TYPE);
236  } else {
237  setWindowFlags(windowFlags() | RUBBERBAND_WINDOW_TYPE);
238  }
239  break;
240  default:
241  break;
242  }
243 
244  if (e->type() == QEvent::ZOrderChange)
245  raise();
