69 Q_GUI_EXPORT_INLINE QRgb qRgb(int r, int g, int b)// set RGB value
70 { return (0xffu << 24) | ((r & 0xff) << 16) | ((g & 0xff) << 8) | (b & 0xff); }

72 Q_GUI_EXPORT_INLINE QRgb qRgba(int r, int g, int b, int a)// set RGBA value
73 { return ((a & 0xff) << 24) | ((r & 0xff) << 16) | ((g & 0xff) << 8) | (b & 0xff); }

75 Q_GUI_EXPORT_INLINE int qGray(int r, int g, int b)// convert R,G,B to gray 0..255

78 Q_GUI_EXPORT_INLINE int qGray(QRgb rgb) // convert RGB to gray 0..255
