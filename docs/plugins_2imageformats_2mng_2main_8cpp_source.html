70 QImageIOPlugin::Capabilities QMngPlugin::capabilities(QIODevice *device, const QByteArray &format) const
71 {
72  if (format == "mng")
73  return Capabilities(CanRead);
74  if (!format.isEmpty())
75  return 0;
76  if (!device->isOpen())
77  return 0;
78 
79  Capabilities cap;
80  if (device->isReadable() && QMngHandler::canRead(device))
81  cap |= CanRead;
82  return cap;
83 }
