475 void QAbstractTextDocumentLayout::registerHandler(int formatType, QObject *component)
476 {
477  Q_D(QAbstractTextDocumentLayout);
478 
479  QTextObjectInterface *iface = qobject_cast<QTextObjectInterface *>(component);
480  if (!iface)
481  return; // ### print error message on terminal?

516 void QAbstractTextDocumentLayout::resizeInlineObject(QTextInlineObject item, int posInDocument, const QTextFormat &format)
517 {
518  Q_D(QAbstractTextDocumentLayout);
519 
520  QTextCharFormat f = format.toCharFormat();
521  Q_ASSERT(f.isValid());
522  QTextObjectHandler handler = d->handlers.value(f.objectType());
523  if (!handler.component)
524  return;
525 
526  QSizeF s = handler.iface->intrinsicSize(document(), posInDocument, format);
527  item.setWidth(s.width());
528  item.setAscent(s.height() - 1);
