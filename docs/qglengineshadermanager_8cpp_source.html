299 QGLEngineShaderProg *QGLEngineSharedShaders::findProgramInCache(const QGLEngineShaderProg &prog)
300 {
301  for (int i = 0; i < cachedPrograms.size(); ++i) {
302  QGLEngineShaderProg *cachedProg = cachedPrograms[i];
303  if (*cachedProg == prog) {
304  // Move the program to the top of the list as a poor-man's cache algo
305  cachedPrograms.move(i, 0);
306  cachedProg->program->bind();
307  return cachedProg;
308  }
309  }
310 
311  QScopedPointer<QGLEngineShaderProg> newProg;
312 
313  do {
314  QByteArray fragSource;
315  // Insert the custom stage before the srcPixel shader to work around an ATI driver bug
316  // where you cannot forward declare a function that takes a sampler as argument.
317  if (prog.srcPixelFragShader == CustomImageSrcFragmentShader)
318  fragSource.append(prog.customStageSource);
319  fragSource.append(qShaderSnippets[prog.mainFragShader]);
320  fragSource.append(qShaderSnippets[prog.srcPixelFragShader]);
321  if (prog.compositionFragShader)
322  fragSource.append(qShaderSnippets[prog.compositionFragShader]);
323  if (prog.maskFragShader)
324  fragSource.append(qShaderSnippets[prog.maskFragShader]);
325 
326  QByteArray vertexSource;
327  vertexSource.append(qShaderSnippets[prog.mainVertexShader]);
328  vertexSource.append(qShaderSnippets[prog.positionVertexShader]);
329 
330  QScopedPointer<QGLShaderProgram> shaderProgram(new QGLShaderProgram(ctxGuard.context(), 0));
331 
332  CachedShader shaderCache(fragSource, vertexSource);
333  bool inCache = shaderCache.load(shaderProgram.data(), ctxGuard.context());
334 
335  if (!inCache) {
336 
337  QScopedPointer<QGLShader> fragShader(new QGLShader(QGLShader::Fragment, ctxGuard.context(), 0));
338  QByteArray description;
339 #if defined(QT_DEBUG)
340  // Name the shader for easier debugging
341  description.append("Fragment shader: main=");
342  description.append(snippetNameStr(prog.mainFragShader));
343  description.append(", srcPixel=");
344  description.append(snippetNameStr(prog.srcPixelFragShader));
345  if (prog.compositionFragShader) {
346  description.append(", composition=");
347  description.append(snippetNameStr(prog.compositionFragShader));
348  }
349  if (prog.maskFragShader) {
350  description.append(", mask=");
351  description.append(snippetNameStr(prog.maskFragShader));
352  }
353  fragShader->setObjectName(QString::fromLatin1(description));
354 #endif
355  if (!fragShader->compileSourceCode(fragSource)) {
356  qWarning() << "Warning:" << description << "failed to compile!";
357  break;
358  }
359 
360  QScopedPointer<QGLShader> vertexShader(new QGLShader(QGLShader::Vertex, ctxGuard.context(), 0));
361 #if defined(QT_DEBUG)
362  // Name the shader for easier debugging
363  description.clear();
364  description.append("Vertex shader: main=");
365  description.append(snippetNameStr(prog.mainVertexShader));
366  description.append(", position=");
367  description.append(snippetNameStr(prog.positionVertexShader));
368  vertexShader->setObjectName(QString::fromLatin1(description));
369 #endif
370  if (!vertexShader->compileSourceCode(vertexSource)) {
371  qWarning() << "Warning:" << description << "failed to compile!";
372  break;
373  }
374 
375  shaders.append(vertexShader.data());
376  shaders.append(fragShader.data());
377  shaderProgram->addShader(vertexShader.take());
378  shaderProgram->addShader(fragShader.take());
379 
380  // We have to bind the vertex attribute names before the program is linked:
381  shaderProgram->bindAttributeLocation("vertexCoordsArray", QT_VERTEX_COORDS_ATTR);
382  if (prog.useTextureCoords)
383  shaderProgram->bindAttributeLocation("textureCoordArray", QT_TEXTURE_COORDS_ATTR);
384  if (prog.useOpacityAttribute)
385  shaderProgram->bindAttributeLocation("opacityArray", QT_OPACITY_ATTR);
386  if (prog.usePmvMatrixAttribute) {
387  shaderProgram->bindAttributeLocation("pmvMatrix1", QT_PMV_MATRIX_1_ATTR);
388  shaderProgram->bindAttributeLocation("pmvMatrix2", QT_PMV_MATRIX_2_ATTR);
389  shaderProgram->bindAttributeLocation("pmvMatrix3", QT_PMV_MATRIX_3_ATTR);
390  }
391  }
392 
393  newProg.reset(new QGLEngineShaderProg(prog));
394  newProg->program = shaderProgram.take();
395 
396  newProg->program->link();
397  if (newProg->program->isLinked()) {
398  if (!inCache)
399  shaderCache.store(newProg->program, ctxGuard.context());
400  } else {
401  QLatin1String none("none");
402  QLatin1String br("\n");
403  QString error;
404  error = QLatin1String("Shader program failed to link,");
405 #if defined(QT_DEBUG)
406  error += QLatin1String("\n Shaders Used:\n");
407  for (int i = 0; i < newProg->program->shaders().count(); ++i) {
408  QGLShader *shader = newProg->program->shaders().at(i);
409  error += QLatin1String(" ") + shader->objectName() + QLatin1String(": \n")
410  + QLatin1String(shader->sourceCode()) + br;
411  }
412 #endif
413  error += QLatin1String(" Error Log:\n")
414  + QLatin1String(" ") + newProg->program->log();
415  qWarning() << error;
416  break;
417  }
418 
419  newProg->program->bind();
420 
421  if (newProg->maskFragShader != QGLEngineSharedShaders::NoMaskFragmentShader) {
422  GLuint location = newProg->program->uniformLocation("maskTexture");
423  newProg->program->setUniformValue(location, QT_MASK_TEXTURE_UNIT);
424  }
425 
426  if (cachedPrograms.count() > 30) {
427  // The cache is full, so delete the last 5 programs in the list.
428  // These programs will be least used, as a program us bumped to
429  // the top of the list when it's used.
430  for (int i = 0; i < 5; ++i) {
431  delete cachedPrograms.last();
432  cachedPrograms.removeLast();
433  }
434  }
435 
436  cachedPrograms.insert(0, newProg.data());
437  } while (false);
438 
439  return newProg.take();
440 }

625 // Select & use the correct shader program using the current state.
626 // Returns true if program needed changing.
627 bool QGLEngineShaderManager::useCorrectShaderProg()
628 {
629  if (!shaderProgNeedsChanging)
630  return false;
631 
632  bool useCustomSrc = customSrcStage != 0;
633  if (useCustomSrc && srcPixelType != QGLEngineShaderManager::ImageSrc && srcPixelType != Qt::TexturePattern) {
634  useCustomSrc = false;
635  qWarning("QGLEngineShaderManager - Ignoring custom shader stage for non image src");
636  }
637 
638  QGLEngineShaderProg requiredProgram;
639 
640  bool texCoords = false;
641 
642  // Choose vertex shader shader position function (which typically also sets
643  // varyings) and the source pixel (srcPixel) fragment shader function:
644  requiredProgram.positionVertexShader = QGLEngineSharedShaders::InvalidSnippetName;
645  requiredProgram.srcPixelFragShader = QGLEngineSharedShaders::InvalidSnippetName;
646  bool isAffine = brushTransform.isAffine();
647  if ( (srcPixelType >= Qt::Dense1Pattern) && (srcPixelType <= Qt::DiagCrossPattern) ) {
648  if (isAffine)
649  requiredProgram.positionVertexShader = QGLEngineSharedShaders::AffinePositionWithPatternBrushVertexShader;
650  else
651  requiredProgram.positionVertexShader = QGLEngineSharedShaders::PositionWithPatternBrushVertexShader;
652 
653  requiredProgram.srcPixelFragShader = QGLEngineSharedShaders::PatternBrushSrcFragmentShader;
654  }
655  else switch (srcPixelType) {
656  default:
657  case Qt::NoBrush:
658  qFatal("QGLEngineShaderManager::useCorrectShaderProg() - Qt::NoBrush style is set");
659  break;
660  case QGLEngineShaderManager::ImageSrc:
661  requiredProgram.srcPixelFragShader = QGLEngineSharedShaders::ImageSrcFragmentShader;
662  requiredProgram.positionVertexShader = QGLEngineSharedShaders::PositionOnlyVertexShader;
663  texCoords = true;
664  break;
665  case QGLEngineShaderManager::NonPremultipliedImageSrc:
666  requiredProgram.srcPixelFragShader = QGLEngineSharedShaders::NonPremultipliedImageSrcFragmentShader;
667  requiredProgram.positionVertexShader = QGLEngineSharedShaders::PositionOnlyVertexShader;
668  texCoords = true;
669  break;
670  case QGLEngineShaderManager::PatternSrc:
671  requiredProgram.srcPixelFragShader = QGLEngineSharedShaders::ImageSrcWithPatternFragmentShader;
672  requiredProgram.positionVertexShader = QGLEngineSharedShaders::PositionOnlyVertexShader;
673  texCoords = true;
674  break;
675  case QGLEngineShaderManager::TextureSrcWithPattern:
676  requiredProgram.srcPixelFragShader = QGLEngineSharedShaders::TextureBrushSrcWithPatternFragmentShader;
677  requiredProgram.positionVertexShader = isAffine ? QGLEngineSharedShaders::AffinePositionWithTextureBrushVertexShader
678  : QGLEngineSharedShaders::PositionWithTextureBrushVertexShader;
679  break;
680  case Qt::SolidPattern:
681  requiredProgram.srcPixelFragShader = QGLEngineSharedShaders::SolidBrushSrcFragmentShader;
682  requiredProgram.positionVertexShader = QGLEngineSharedShaders::PositionOnlyVertexShader;
683  break;
684  case Qt::LinearGradientPattern:
685  requiredProgram.srcPixelFragShader = QGLEngineSharedShaders::LinearGradientBrushSrcFragmentShader;
686  requiredProgram.positionVertexShader = isAffine ? QGLEngineSharedShaders::AffinePositionWithLinearGradientBrushVertexShader
687  : QGLEngineSharedShaders::PositionWithLinearGradientBrushVertexShader;
688  break;
689  case Qt::ConicalGradientPattern:
690  requiredProgram.srcPixelFragShader = QGLEngineSharedShaders::ConicalGradientBrushSrcFragmentShader;
691  requiredProgram.positionVertexShader = isAffine ? QGLEngineSharedShaders::AffinePositionWithConicalGradientBrushVertexShader
692  : QGLEngineSharedShaders::PositionWithConicalGradientBrushVertexShader;
693  break;
694  case Qt::RadialGradientPattern:
695  requiredProgram.srcPixelFragShader = QGLEngineSharedShaders::RadialGradientBrushSrcFragmentShader;
696  requiredProgram.positionVertexShader = isAffine ? QGLEngineSharedShaders::AffinePositionWithRadialGradientBrushVertexShader
697  : QGLEngineSharedShaders::PositionWithRadialGradientBrushVertexShader;
698  break;
699  case Qt::TexturePattern:
700  requiredProgram.srcPixelFragShader = QGLEngineSharedShaders::TextureBrushSrcFragmentShader;
701  requiredProgram.positionVertexShader = isAffine ? QGLEngineSharedShaders::AffinePositionWithTextureBrushVertexShader
702  : QGLEngineSharedShaders::PositionWithTextureBrushVertexShader;
703  break;
704  };
705 
706  if (useCustomSrc) {
707  requiredProgram.srcPixelFragShader = QGLEngineSharedShaders::CustomImageSrcFragmentShader;
708  requiredProgram.customStageSource = customSrcStage->source();
709  }
710 
711  const bool hasCompose = compositionMode > QPainter::CompositionMode_Plus;
712  const bool hasMask = maskType != QGLEngineShaderManager::NoMask;
713 
714  // Choose fragment shader main function:
715  if (opacityMode == AttributeOpacity) {
716  Q_ASSERT(!hasCompose && !hasMask);
717  requiredProgram.mainFragShader = QGLEngineSharedShaders::MainFragmentShader_ImageArrays;
718  } else {
719  bool useGlobalOpacity = (opacityMode == UniformOpacity);
720  if (hasCompose && hasMask && useGlobalOpacity)
721  requiredProgram.mainFragShader = QGLEngineSharedShaders::MainFragmentShader_CMO;
722  if (hasCompose && hasMask && !useGlobalOpacity)
723  requiredProgram.mainFragShader = QGLEngineSharedShaders::MainFragmentShader_CM;
724  if (!hasCompose && hasMask && useGlobalOpacity)
725  requiredProgram.mainFragShader = QGLEngineSharedShaders::MainFragmentShader_MO;
726  if (!hasCompose && hasMask && !useGlobalOpacity)
727  requiredProgram.mainFragShader = QGLEngineSharedShaders::MainFragmentShader_M;
728  if (hasCompose && !hasMask && useGlobalOpacity)
729  requiredProgram.mainFragShader = QGLEngineSharedShaders::MainFragmentShader_CO;
730  if (hasCompose && !hasMask && !useGlobalOpacity)
731  requiredProgram.mainFragShader = QGLEngineSharedShaders::MainFragmentShader_C;
732  if (!hasCompose && !hasMask && useGlobalOpacity)
733  requiredProgram.mainFragShader = QGLEngineSharedShaders::MainFragmentShader_O;
734  if (!hasCompose && !hasMask && !useGlobalOpacity)
735  requiredProgram.mainFragShader = QGLEngineSharedShaders::MainFragmentShader;
736  }
737 
738  if (hasMask) {
739  if (maskType == PixelMask) {
740  requiredProgram.maskFragShader = QGLEngineSharedShaders::MaskFragmentShader;
741  texCoords = true;
742  } else if (maskType == SubPixelMaskPass1) {
743  requiredProgram.maskFragShader = QGLEngineSharedShaders::RgbMaskFragmentShaderPass1;
744  texCoords = true;
745  } else if (maskType == SubPixelMaskPass2) {
746  requiredProgram.maskFragShader = QGLEngineSharedShaders::RgbMaskFragmentShaderPass2;
747  texCoords = true;
748  } else if (maskType == SubPixelWithGammaMask) {
749  requiredProgram.maskFragShader = QGLEngineSharedShaders::RgbMaskWithGammaFragmentShader;
750  texCoords = true;
751  } else {
752  qCritical("QGLEngineShaderManager::useCorrectShaderProg() - Unknown mask type");
753  }
754  } else {
755  requiredProgram.maskFragShader = QGLEngineSharedShaders::NoMaskFragmentShader;
756  }
757 
758  if (hasCompose) {
759  switch (compositionMode) {
760  case QPainter::CompositionMode_Multiply:
761  requiredProgram.compositionFragShader = QGLEngineSharedShaders::MultiplyCompositionModeFragmentShader;
762  break;
763  case QPainter::CompositionMode_Screen:
764  requiredProgram.compositionFragShader = QGLEngineSharedShaders::ScreenCompositionModeFragmentShader;
765  break;
766  case QPainter::CompositionMode_Overlay:
767  requiredProgram.compositionFragShader = QGLEngineSharedShaders::OverlayCompositionModeFragmentShader;
768  break;
769  case QPainter::CompositionMode_Darken:
770  requiredProgram.compositionFragShader = QGLEngineSharedShaders::DarkenCompositionModeFragmentShader;
771  break;
772  case QPainter::CompositionMode_Lighten:
773  requiredProgram.compositionFragShader = QGLEngineSharedShaders::LightenCompositionModeFragmentShader;
774  break;
775  case QPainter::CompositionMode_ColorDodge:
776  requiredProgram.compositionFragShader = QGLEngineSharedShaders::ColorDodgeCompositionModeFragmentShader;
777  break;
778  case QPainter::CompositionMode_ColorBurn:
779  requiredProgram.compositionFragShader = QGLEngineSharedShaders::ColorBurnCompositionModeFragmentShader;
780  break;
781  case QPainter::CompositionMode_HardLight:
782  requiredProgram.compositionFragShader = QGLEngineSharedShaders::HardLightCompositionModeFragmentShader;
783  break;
784  case QPainter::CompositionMode_SoftLight:
785  requiredProgram.compositionFragShader = QGLEngineSharedShaders::SoftLightCompositionModeFragmentShader;
786  break;
787  case QPainter::CompositionMode_Difference:
788  requiredProgram.compositionFragShader = QGLEngineSharedShaders::DifferenceCompositionModeFragmentShader;
789  break;
790  case QPainter::CompositionMode_Exclusion:
791  requiredProgram.compositionFragShader = QGLEngineSharedShaders::ExclusionCompositionModeFragmentShader;
792  break;
793  default:
794  qWarning("QGLEngineShaderManager::useCorrectShaderProg() - Unsupported composition mode");
795  }
796  } else {
797  requiredProgram.compositionFragShader = QGLEngineSharedShaders::NoCompositionModeFragmentShader;
798  }
799 
800  // Choose vertex shader main function
801  if (opacityMode == AttributeOpacity) {
802  Q_ASSERT(texCoords);
803  requiredProgram.mainVertexShader = QGLEngineSharedShaders::MainWithTexCoordsAndOpacityVertexShader;
804  } else if (texCoords) {
805  requiredProgram.mainVertexShader = QGLEngineSharedShaders::MainWithTexCoordsVertexShader;
806  } else {
807  requiredProgram.mainVertexShader = QGLEngineSharedShaders::MainVertexShader;
808  }
809  requiredProgram.useTextureCoords = texCoords;
810  requiredProgram.useOpacityAttribute = (opacityMode == AttributeOpacity);
811  if (complexGeometry && srcPixelType == Qt::SolidPattern) {
812  requiredProgram.positionVertexShader = QGLEngineSharedShaders::ComplexGeometryPositionOnlyVertexShader;
813  requiredProgram.usePmvMatrixAttribute = false;
814  } else {
815  requiredProgram.usePmvMatrixAttribute = true;
816 
817  // Force complexGeometry off, since we currently don't support that mode for
818  // non-solid brushes
819  complexGeometry = false;
820  }
821 
822  // At this point, requiredProgram is fully populated so try to find the program in the cache
823  currentShaderProg = sharedShaders->findProgramInCache(requiredProgram);
824 
825  if (currentShaderProg && useCustomSrc) {
826  customSrcStage->setUniforms(currentShaderProg->program);
827  }
828 
829  // Make sure all the vertex attribute arrays the program uses are enabled (and the ones it
830  // doesn't use are disabled)
831  QGLContextPrivate* ctx_d = ctx->d_func();
832  ctx_d->setVertexAttribArrayEnabled(QT_VERTEX_COORDS_ATTR, true);
833  ctx_d->setVertexAttribArrayEnabled(QT_TEXTURE_COORDS_ATTR, currentShaderProg && currentShaderProg->useTextureCoords);
834  ctx_d->setVertexAttribArrayEnabled(QT_OPACITY_ATTR, currentShaderProg && currentShaderProg->useOpacityAttribute);
835 
836  shaderProgNeedsChanging = false;
837  return true;
838 }
