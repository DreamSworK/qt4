195 bool QBasicMouseEventTransition::eventTest(QEvent *event)
196 {
197  Q_D(const QBasicMouseEventTransition);
198  if (event->type() == d->eventType) {
199  QMouseEvent *me = static_cast<QMouseEvent*>(event);
200  return (me->button() == d->button)
201  && ((me->modifiers() & d->modifierMask) == d->modifierMask)
202  && (d->path.isEmpty() || d->path.contains(me->pos()));
203  }
204  return false;
205 }
