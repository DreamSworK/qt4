639 bool QUrlInfo::greaterThan(const QUrlInfo &i1, const QUrlInfo &i2,
640  int sortBy)
641 {
642  switch (sortBy) {
643  case QDir::Name:
644  return i1.name() > i2.name();
645  case QDir::Time:
646  return i1.lastModified() > i2.lastModified();
647  case QDir::Size:
648  return i1.size() > i2.size();
649  default:
650  return false;
651  }
652 }
