251 QUrl QDeclarativeFolderListModel::parentFolder() const
252 {
253  QString localFile = d->folder.toLocalFile();
254  if (!localFile.isEmpty()) {
255  QDir dir(localFile);
256 #if defined(Q_OS_SYMBIAN) || defined(Q_OS_WIN)
257  if (dir.isRoot())
258  dir.setPath("");
259  else
260 #endif
261  dir.cdUp();
262  localFile = dir.path();
263  } else {
264  int pos = d->folder.path().lastIndexOf(QLatin1Char('/'));
265  if (pos == -1)
266  return QUrl();
267  localFile = d->folder.path().left(pos);
268  }
269  return QUrl::fromLocalFile(localFile);
270 }

385 void QDeclarativeFolderListModel::refresh()
386 {
387  d->folderIndex = QModelIndex();
388  if (d->count) {
389  emit beginRemoveRows(QModelIndex(), 0, d->count-1);
390  d->count = 0;
391  emit endRemoveRows();
392  }
393  d->folderIndex = d->model.index(d->folder.toLocalFile());
394  int newcount = d->model.rowCount(d->folderIndex);
395  if (newcount) {
396  emit beginInsertRows(QModelIndex(), 0, newcount-1);
397  d->count = newcount;
398  emit endInsertRows();
399  }
