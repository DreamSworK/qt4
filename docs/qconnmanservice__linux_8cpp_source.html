96 void QConnmanManagerInterface::connectNotify(const char *signal)
97 {
98 if (QLatin1String(signal) == SIGNAL(propertyChanged(QString,QDBusVariant))) {
99  if(!connection().connect(QLatin1String(CONNMAN_SERVICE),
100  QLatin1String(CONNMAN_MANAGER_PATH),
101  QLatin1String(CONNMAN_MANAGER_INTERFACE),
102  QLatin1String("PropertyChanged"),
103  this,SIGNAL(propertyChanged(const QString &, const QDBusVariant & )))) {
104  qWarning() << "PropertyCHanged not connected";
105  }
106  }
107 
108  if (QLatin1String(signal) == SIGNAL(stateChanged(QString))) {
109  if (!connection().connect(QLatin1String(CONNMAN_SERVICE),
110  QLatin1String(CONNMAN_MANAGER_PATH),
111  QLatin1String(CONNMAN_MANAGER_INTERFACE),
112  QLatin1String("StateChanged"),
113  this,SIGNAL(stateChanged(const QString&)))) {
114  qWarning() << "StateChanged not connected";
115 
116  }
117  }
118  if (QLatin1String(signal) == SIGNAL(propertyChangedContext(QString,QString,QDBusVariant))) {
119  QConnmanDBusHelper *helper;
120  helper = new QConnmanDBusHelper(this);
121 
122  dbusConnection.connect(QLatin1String(CONNMAN_SERVICE),
123  QLatin1String(CONNMAN_MANAGER_PATH),
124  QLatin1String(CONNMAN_MANAGER_INTERFACE),
125  QLatin1String("PropertyChanged"),
126  helper,SLOT(propertyChanged(QString,QDBusVariant)));
127 
128 
129  QObject::connect(helper,SIGNAL(propertyChangedContext(const QString &,const QString &,const QDBusVariant &)),
130  this,SIGNAL(propertyChangedContext(const QString &,const QString &,const QDBusVariant &)), Qt::UniqueConnection);
131  }
132 }
