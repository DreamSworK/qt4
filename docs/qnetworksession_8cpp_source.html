261 QNetworkSession::QNetworkSession(const QNetworkConfiguration &connectionConfig, QObject *parent)
262  : QObject(parent), d(0)
263 {
264  // invalid configuration
265  if (!connectionConfig.identifier().isEmpty()) {
266  foreach (QBearerEngine *engine, qNetworkConfigurationManagerPrivate()->engines()) {
267  if (engine->hasIdentifier(connectionConfig.identifier())) {
268  d = engine->createSessionBackend();
269  d->q = this;
270  d->publicConfig = connectionConfig;
271  d->syncStateWithInterface();
272  connect(d, SIGNAL(quitPendingWaitsForOpened()), this, SIGNAL(opened()));
273  connect(d, SIGNAL(error(QNetworkSession::SessionError)),
274  this, SIGNAL(error(QNetworkSession::SessionError)));
275  connect(d, SIGNAL(stateChanged(QNetworkSession::State)),
276  this, SIGNAL(stateChanged(QNetworkSession::State)));
277  connect(d, SIGNAL(closed()), this, SIGNAL(closed()));
278  connect(d, SIGNAL(preferredConfigurationChanged(QNetworkConfiguration,bool)),
279  this, SIGNAL(preferredConfigurationChanged(QNetworkConfiguration,bool)));
280  connect(d, SIGNAL(newConfigurationActivated()),
281  this, SIGNAL(newConfigurationActivated()));
282  break;
283  }
284  }
285  }

341 bool QNetworkSession::waitForOpened(int msecs)
342 {
343  if (!d)
344  return false;
345 
346  if (d->isOpen)
347  return true;
348 
349  if (!(d->state == Connecting || d->state == Connected)) {
350  return false;
351  }
352 
353  QEventLoop loop;
354  QObject::connect(d, SIGNAL(quitPendingWaitsForOpened()), &loop, SLOT(quit()));
355  QObject::connect(this, SIGNAL(error(QNetworkSession::SessionError)), &loop, SLOT(quit()));
356 
357  //final call
358  if (msecs >= 0)
359  QTimer::singleShot(msecs, &loop, SLOT(quit()));
360 
361  // enter the event loop and wait for opened/error/timeout
362  loop.exec(QEventLoop::ExcludeUserInputEvents | QEventLoop::WaitForMoreEvents);
363 
364  return d->isOpen;
365 }

553 QVariant QNetworkSession::sessionProperty(const QString &key) const
554 {
555  if (!d || !d->publicConfig.isValid())
556  return QVariant();
557 
558  if (key == QLatin1String("ActiveConfiguration"))
559  return d->isOpen ? d->activeConfig.identifier() : QString();
560 
561  if (key == QLatin1String("UserChoiceConfiguration")) {
562  if (!d->isOpen || d->publicConfig.type() != QNetworkConfiguration::UserChoice)
563  return QString();
564 
565  if (d->serviceConfig.isValid())
566  return d->serviceConfig.identifier();
567  else
568  return d->activeConfig.identifier();
569  }
570 
571  return d->sessionProperty(key);
572 }
