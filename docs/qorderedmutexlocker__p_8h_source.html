60 /*
61  Locks 2 mutexes in a defined order, avoiding a recursive lock if
62  we're trying to lock the same mutex twice.
63 */
64 class QOrderedMutexLocker
65 {
66 public:
67  QOrderedMutexLocker(QMutex *m1, QMutex *m2)
68  : mtx1((m1 == m2) ? m1 : (m1 < m2 ? m1 : m2)),
69  mtx2((m1 == m2) ? 0 : (m1 < m2 ? m2 : m1)),
70  locked(false)
71  {
72  relock();
73  }
74  ~QOrderedMutexLocker()
75  {
76  unlock();
77  }
78 
79  void relock()
80  {
81  if (!locked) {
82  if (mtx1) mtx1->lockInline();
83  if (mtx2) mtx2->lockInline();
84  locked = true;
85  }
86  }
87 
88  void unlock()
89  {
90  if (locked) {
91  if (mtx1) mtx1->unlockInline();
92  if (mtx2) mtx2->unlockInline();
93  locked = false;
94  }
95  }
96 
97  static bool relock(QMutex *mtx1, QMutex *mtx2)
98  {
99  // mtx1 is already locked, mtx2 not... do we need to unlock and relock?
100  if (mtx1 == mtx2)
101  return false;
102  if (mtx1 < mtx2) {
103  mtx2->lockInline();
104  return true;
105  }
106  if (!mtx2->tryLockInline()) {
107  mtx1->unlock();
108  mtx2->lock();
109  mtx1->lock();
110  }
111  return true;
112  }
113 
114 private:
115  QMutex *mtx1, *mtx2;
116  bool locked;
117 };
