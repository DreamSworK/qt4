50 static bool sendWindowSystemEvents(QAbstractEventDispatcher *eventDispatcher, QEventLoop::ProcessEventsFlags flags)
51 {
52  int nevents = 0;
53 
54  // handle gui and posted events
55  QCoreApplication::sendPostedEvents();
56 
57  while (true) {
58  QWindowSystemInterfacePrivate::WindowSystemEvent *event;
59  if (!(flags & QEventLoop::ExcludeUserInputEvents)
60  && QWindowSystemInterfacePrivate::windowSystemEventsQueued() > 0) {
61  // process a pending user input event
62  event = QWindowSystemInterfacePrivate::getWindowSystemEvent();
63  if (!event)
64  break;
65  } else {
66  break;
67  }
68 
69  // in contrast to the common QPA version we don't put window system events
70  // into the event filter here, instead native bps events are filtered in
71  // QEventDispatcherBlackberry::select()
72 
73  nevents++;
74 
75  QApplicationPrivate::processWindowSystemEvent(event);
76  delete event;
77  }
78 
79  return (nevents > 0);
80 }

92 {
