72 void AbstractViewInspector::createQmlObject(const QString &qml, QObject *parent,
73  const QStringList &importList,
74  const QString &filename)
75 {
76  if (!parent)
77  return;
78 
79  QString imports;
80  foreach (const QString &s, importList) {
81  imports += s;
82  imports += QLatin1Char('\n');
83  }
84 
85  QDeclarativeContext *parentContext = declarativeEngine()->contextForObject(parent);
86  QDeclarativeComponent component(declarativeEngine());
87  QByteArray constructedQml = QString(imports + qml).toLatin1();
88 
89  component.setData(constructedQml, QUrl::fromLocalFile(filename));
90  QObject *newObject = component.create(parentContext);
91  if (newObject)
92  reparentQmlObject(newObject, parent);
93 }

191 bool AbstractViewInspector::eventFilter(QObject *obj, QEvent *event)
192 {
193  if (!designModeBehavior())
194  return QObject::eventFilter(obj, event);
195 
196  switch (event->type()) {
197  case QEvent::Leave:
198  if (leaveEvent(event))
199  return true;
200  break;
201  case QEvent::MouseButtonPress:
202  if (mousePressEvent(static_cast<QMouseEvent*>(event)))
203  return true;
204  break;
205  case QEvent::MouseMove:
206  if (mouseMoveEvent(static_cast<QMouseEvent*>(event)))
207  return true;
208  break;
209  case QEvent::MouseButtonRelease:
210  if (mouseReleaseEvent(static_cast<QMouseEvent*>(event)))
211  return true;
212  break;
213  case QEvent::KeyPress:
214  if (keyPressEvent(static_cast<QKeyEvent*>(event)))
215  return true;
216  break;
217  case QEvent::KeyRelease:
218  if (keyReleaseEvent(static_cast<QKeyEvent*>(event)))
219  return true;
220  break;
221  case QEvent::MouseButtonDblClick:
222  if (mouseDoubleClickEvent(static_cast<QMouseEvent*>(event)))
223  return true;
224  break;
225  case QEvent::Wheel:
226  if (wheelEvent(static_cast<QWheelEvent*>(event)))
227  return true;
228  break;
229  default:
230  break;
231  }
232 
233  return QObject::eventFilter(obj, event);
234 }

309 void AbstractViewInspector::handleMessage(const QByteArray &message)
310 {
311  QDataStream ds(message);
312 
313  InspectorProtocol::Message type;
314  ds >> type;
315 
316  switch (type) {
317  case InspectorProtocol::SetCurrentObjects: {
318  int itemCount = 0;
319  ds >> itemCount;
320 
321  QList<QObject*> selectedObjects;
322  for (int i = 0; i < itemCount; ++i) {
323  int debugId = -1;
324  ds >> debugId;
325  if (QObject *obj = QDeclarativeDebugService::objectForId(debugId))
326  selectedObjects << obj;
327  }
328 
329  changeCurrentObjects(selectedObjects);
330  break;
331  }
332  case InspectorProtocol::Reload: {
333  reloadView();
334  break;
335  }
336  case InspectorProtocol::SetAnimationSpeed: {
337  qreal speed;
338  ds >> speed;
339  animationSpeedChangeRequested(speed);
340  break;
341  }
342  case InspectorProtocol::SetAnimationPaused: {
343  bool paused;
344  ds >> paused;
345  animationPausedChangeRequested(paused);
346  break;
347  }
348  case InspectorProtocol::ChangeTool: {
349  InspectorProtocol::Tool tool;
350  ds >> tool;
351  changeTool(tool);
352  break;
353  }
354  case InspectorProtocol::SetDesignMode: {
355  bool inDesignMode;
356  ds >> inDesignMode;
357  setDesignModeBehavior(inDesignMode);
358  break;
359  }
360  case InspectorProtocol::ShowAppOnTop: {
361  bool showOnTop;
362  ds >> showOnTop;
363  setShowAppOnTop(showOnTop);
364  break;
365  }
366  case InspectorProtocol::CreateObject: {
367  QString qml;
368  int parentId;
369  QString filename;
370  QStringList imports;
371  ds >> qml >> parentId >> imports >> filename;
372  createQmlObject(qml, QDeclarativeDebugService::objectForId(parentId),
373  imports, filename);
374  break;
375  }
376  case InspectorProtocol::DestroyObject: {
377  int debugId;
378  ds >> debugId;
379  if (QObject *obj = QDeclarativeDebugService::objectForId(debugId))
380  obj->deleteLater();
381  break;
382  }
383  case InspectorProtocol::MoveObject: {
384  int debugId, newParent;
385  ds >> debugId >> newParent;
386  reparentQmlObject(QDeclarativeDebugService::objectForId(debugId),
387  QDeclarativeDebugService::objectForId(newParent));
388  break;
389  }
390  case InspectorProtocol::ObjectIdList: {
391  int itemCount;
392  ds >> itemCount;
393  m_stringIdForObjectId.clear();
394  for (int i = 0; i < itemCount; ++i) {
395  int itemDebugId;
396  QString itemIdString;
397  ds >> itemDebugId
398  >> itemIdString;
399 
400  m_stringIdForObjectId.insert(itemDebugId, itemIdString);
401  }
402  break;
403  }
404  case InspectorProtocol::ClearComponentCache: {
405  clearComponentCache();
406  break;
407  }
408  default:
409  qWarning() << "Warning: Not handling message:" << type;
410  }
411 }
