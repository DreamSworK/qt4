203 void QStyleOption::init(const QWidget *widget)
204 {
205  QWidget *window = widget->window();
206  state = QStyle::State_None;
207  if (widget->isEnabled())
208  state |= QStyle::State_Enabled;
209  if (widget->hasFocus())
210  state |= QStyle::State_HasFocus;
211  if (window->testAttribute(Qt::WA_KeyboardFocusChange))
212  state |= QStyle::State_KeyboardFocusChange;
213  if (widget->underMouse())
214  state |= QStyle::State_MouseOver;
215  if (window->isActiveWindow())
216  state |= QStyle::State_Active;
217  if (widget->isWindow())
218  state |= QStyle::State_Window;
219 #ifdef Q_WS_MAC
220  extern bool qt_mac_can_clickThrough(const QWidget *w); //qwidget_mac.cpp
221  if (!(state & QStyle::State_Active) && !qt_mac_can_clickThrough(widget))
222  state &= ~QStyle::State_Enabled;
223 
224  switch (QMacStyle::widgetSizePolicy(widget)) {
225  case QMacStyle::SizeSmall:
226  state |= QStyle::State_Small;
227  break;
228  case QMacStyle::SizeMini:
229  state |= QStyle::State_Mini;
230  break;
231  default:
232  ;
233  }
234 #endif
235 #ifdef QT_KEYPAD_NAVIGATION
236  if (widget->hasEditFocus())
237  state |= QStyle::State_HasEditFocus;
238 #endif
239 
240  direction = widget->layoutDirection();
241  rect = widget->rect();
242  palette = widget->palette();
243  fontMetrics = widget->fontMetrics();
244 }
