379 static inline QDataStream &operator>>(QDataStream &in, QNetworkCacheMetaData::AttributesMap &hash)
380 {
381  hash.clear();
382  QDataStream::Status oldStatus = in.status();
383  in.resetStatus();
384  hash.clear();
385 
386  quint32 n;
387  in >> n;
388 
389  for (quint32 i = 0; i < n; ++i) {
390  if (in.status() != QDataStream::Ok)
391  break;
392 
393  int k;
394  QVariant t;
395  in >> k >> t;
396  hash.insertMulti(QNetworkRequest::Attribute(k), t);
397  }
398 
399  if (in.status() != QDataStream::Ok)
400  hash.clear();
401  if (oldStatus != QDataStream::Ok)
402  in.setStatus(oldStatus);
403  return in;
404 }
