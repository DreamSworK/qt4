54 void GenericNamespaceResolver::addBinding(const QXmlName nb)
55 {
56  if(nb.namespaceURI() == StandardNamespaces::UndeclarePrefix)
57  m_bindings.remove(nb.prefix());
58  else
59  m_bindings.insert(nb.prefix(), nb.namespaceURI());
60 }
