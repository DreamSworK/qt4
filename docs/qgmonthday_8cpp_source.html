54 GMonthDay::Ptr GMonthDay::fromLexical(const QString &lexical)
55 {
56  static const CaptureTable captureTable( // STATIC DATA
57  /* The extra paranthesis is a build fix for GCC 3.3. */
58  (QRegExp(QLatin1String(
59  "^\\s*" /* Any preceding whitespace. */
60  "--" /* Delimiter. */
61  "(\\d{2})" /* The month part. */
62  "-" /* Delimiter. */
63  "(\\d{2})" /* The day part. */
64  "(?:(\\+|-)(\\d{2}):(\\d{2})|(Z))?" /* The zone offset, "+08:24". */
65  "\\s*$" /* Any terminating whitespace. */))),
66  /*zoneOffsetSignP*/ 3,
67  /*zoneOffsetHourP*/ 4,
68  /*zoneOffsetMinuteP*/ 5,
69  /*zoneOffsetUTCSymbolP*/ 6,
70  /*yearP*/ -1,
71  /*monthP*/ 1,
72  /*dayP*/ 2);
73 
74  AtomicValue::Ptr err;
75  const QDateTime retval(create(err, lexical, captureTable));
76 
77  return err ? err : GMonthDay::Ptr(new GMonthDay(retval));
78 }
