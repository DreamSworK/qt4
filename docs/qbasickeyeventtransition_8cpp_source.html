191 bool QBasicKeyEventTransition::eventTest(QEvent *event)
192 {
193  Q_D(const QBasicKeyEventTransition);
194  if (event->type() == d->eventType) {
195  QKeyEvent *ke = static_cast<QKeyEvent*>(event);
196  return (ke->key() == d->key)
197  && ((ke->modifiers() & d->modifierMask) == d->modifierMask);
198  }
199  return false;
200 }
