73  m_thisFBO = view.fbo;

79 #if defined(QT_OPENGL_ES_2)

85  int devType() const { return QInternal::OpenGL; }
86  QSize size() const { return mWindow->geometry().size(); }
87  QGLContext* context() const { return QGLContext::fromPlatformGLContext(mWindow->glContext()); }
