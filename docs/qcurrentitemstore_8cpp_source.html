89 StaticContext::Ptr CurrentItemStore::newStaticContext(const StaticContext::Ptr &context)
90 {
91  /* It might be we are generated despite there is no focus. In that case
92  * an error will reported in case current() is used, but in any case we cannot
93  * crash, so use item() in case we have no focus.
94  *
95  * Such a case is when we're inside a named template, and it's invoked
96  * without focus. */
97  const ItemType::Ptr t(context->contextItemType());
98  return StaticContext::Ptr(new StaticCurrentContext(t ? t : BuiltinTypes::item, context));
99 }

101 Expression::Ptr CurrentItemStore::compress(const StaticContext::Ptr &context)
102 {
103  const Expression::Ptr me(SingleContainer::compress(newStaticContext(context)));
104 
105  if(me != this)
106  return me;
107  else
108  {
109  /* If fn:current() isn't called, there's no point in us sticking
110  * around. */
111  if(m_operand->deepProperties().testFlag(RequiresCurrentItem))
112  return me;
113  else
114  return m_operand;
115  }
116 }
