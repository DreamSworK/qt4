194 template <typename T>
195 inline void qAtomicAssign(T *&d, T *x)
196 {
197  if (d == x)
198  return;
199  x->ref.ref();
200  if (!d->ref.deref())
201  delete d;
202  d = x;
203 }

213 template <typename T>
214 inline void qAtomicDetach(T *&d)
215 {
216  if (d->ref == 1)
217  return;
218  T *x = d;
219  d = new T(*d);
220  if (!x->ref.deref())
221  delete x;
222 }
