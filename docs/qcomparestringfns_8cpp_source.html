55 Item CodepointEqualFN::evaluateSingleton(const DynamicContext::Ptr &context) const
56 {
57  const Item op1(m_operands.first()->evaluateSingleton(context));
58  if(!op1)
59  return Item();
60 
61  const Item op2(m_operands.last()->evaluateSingleton(context));
62  if(!op2)
63  return Item();
64 
65  if(caseSensitivity() == Qt::CaseSensitive)
66  return Boolean::fromValue(op1.stringValue() == op2.stringValue());
67  else
68  {

72  return Boolean::fromValue(s1.length() == s2.length() &&

78 {

80  if(!op1)
81  return Item();
82 
83  const Item op2(m_operands.at(1)->evaluateSingleton(context));
84  if(!op2)
85  return Item();

87  const int retval = caseSensitivity() == Qt::CaseSensitive
88  ? op1.stringValue().compare(op2.stringValue())
89  : op1.stringValue().toLower().compare(op2.stringValue().toLower());
90 
91  if(retval > 0)
92  return CommonValues::IntegerOne;
93  else if(retval < 0)
94  return CommonValues::IntegerOneNegative;
95  else
