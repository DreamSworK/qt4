189 QX11InfoData* QX11Info::getX11Data(bool def) const
190 {
191  QX11InfoData* res = 0;
192  if (def) {
193  res = new QX11InfoData;
194  res->ref = 0;
195  res->screen = appScreen();
196  res->depth = appDepth();
197  res->cells = appCells();
198  res->colormap = colormap();
199  res->defaultColormap = appDefaultColormap();
200  res->visual = (Visual*) appVisual();
201  res->defaultVisual = appDefaultVisual();
202  } else if (x11data) {
203  res = new QX11InfoData;
204  *res = *x11data;
205  res->ref = 0;
206  }
207  return res;
208 }
