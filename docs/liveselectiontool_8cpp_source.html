114 void LiveSelectionTool::mousePressEvent(QMouseEvent *event)
115 {
116  QDeclarativeViewInspectorPrivate *inspectorPrivate

118  QList<QGraphicsItem*> itemList = inspectorPrivate->selectableItems(event->pos());
119  LiveSingleSelectionManipulator::SelectionType selectionType = getSelectionType(event->modifiers());
120 
121  if (event->buttons() & Qt::LeftButton) {
122  m_mousePressTimer.start();
123 
124  if (m_rubberbandSelectionMode) {
125  m_rubberbandSelectionManipulator.begin(event->pos());
126  } else {
127  m_singleSelectionManipulator.begin(event->pos());
128  m_singleSelectionManipulator.select(selectionType, m_selectOnlyContentItems);
129  }
130  } else if (event->buttons() & Qt::RightButton) {
131  createContextMenu(itemList, event->globalPos());

202 void LiveSelectionTool::mouseMoveEvent(QMouseEvent *event)
203 {

205  QPointF mouseMovementVector = m_singleSelectionManipulator.beginPoint() - event->pos();
206 
207  if ((mouseMovementVector.toPoint().manhattanLength() > Constants::DragStartDistance)
208  && (m_mousePressTimer.elapsed() > Constants::DragStartTime))
209  {
210  m_singleSelectionManipulator.end(event->pos());
211  //view()->changeToMoveTool(m_singleSelectionManipulator.beginPoint());

215  QPointF mouseMovementVector = m_rubberbandSelectionManipulator.beginPoint() - event->pos();
216 
217  if ((mouseMovementVector.toPoint().manhattanLength() > Constants::DragStartDistance)
218  && (m_mousePressTimer.elapsed() > Constants::DragStartTime)) {
219  m_rubberbandSelectionManipulator.update(event->pos());
220 
221  if (event->modifiers().testFlag(Qt::ControlModifier))
222  m_rubberbandSelectionManipulator.select(
223  LiveRubberBandSelectionManipulator::RemoveFromSelection);
224  else if (event->modifiers().testFlag(Qt::ShiftModifier))
225  m_rubberbandSelectionManipulator.select(

320 void LiveSelectionTool::wheelEvent(QWheelEvent *event)
321 {
322  if (event->orientation() == Qt::Horizontal || m_rubberbandSelectionMode)
323  return;
324 
325  QDeclarativeViewInspectorPrivate *inspectorPrivate
326  = QDeclarativeViewInspectorPrivate::get(inspector());
327  QList<QGraphicsItem*> itemList = inspectorPrivate->selectableItems(event->pos());
328 
329  if (itemList.isEmpty())
330  return;
331 
332  int selectedIdx = 0;
333  if (!inspector()->selectedItems().isEmpty()) {
334  selectedIdx = itemList.indexOf(inspector()->selectedItems().first());
335  if (selectedIdx >= 0) {
336  if (event->delta() > 0) {
337  selectedIdx++;
338  if (selectedIdx == itemList.length())
339  selectedIdx = 0;
340  } else if (event->delta() < 0) {
341  selectedIdx--;
342  if (selectedIdx == -1)
343  selectedIdx = itemList.length() - 1;
344  }
345  } else {
346  selectedIdx = 0;
347  }
348  }
349 
350  QPointF updatePt(0, 0);
351  m_singleSelectionManipulator.begin(updatePt);
352  m_singleSelectionManipulator.select(LiveSingleSelectionManipulator::ReplaceSelection,
353  QList<QGraphicsItem*>() << itemList.at(selectedIdx),
354  false);
355  m_singleSelectionManipulator.end(updatePt);
356 
357 }
