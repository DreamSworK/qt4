62 QString QDesktopServices::storageLocation(StandardLocation type)
63 {
64  if (type == QDesktopServices::HomeLocation)
65  return QDir::homePath();
66  if (type == QDesktopServices::TempLocation)
67  return QDir::tempPath();
68 
69  if (type == DataLocation) {
70  QString qwsDataHome = QLatin1String(qgetenv("QWS_DATA_HOME"));
71  if (qwsDataHome.isEmpty())
72  qwsDataHome = QDir::homePath() + QLatin1String("/.qws/share");
73  qwsDataHome += QLatin1String("/data/")
74  + QCoreApplication::organizationName() + QLatin1Char('/')
75  + QCoreApplication::applicationName();
76  return qwsDataHome;
77  }
78  if (type == QDesktopServices::CacheLocation) {
79  QString qwsCacheHome = QLatin1String(qgetenv("QWS_CACHE_HOME"));
80  if (qwsCacheHome.isEmpty())
81  qwsCacheHome = QDir::homePath() + QLatin1String("/.qws/cache/");
82  qwsCacheHome += QCoreApplication::organizationName() + QLatin1Char('/')
83  + QCoreApplication::applicationName();
84  return qwsCacheHome;
85  }
86 
87  qWarning("QDesktopServices::storageLocation %d not implemented", type);
88  return QString();
89 }
