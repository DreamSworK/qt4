115 /*
116  Modify total maximum (max), total expansion (exp), and total empty
117  when adding boxmax/boxexp.
118 
119  Expansive boxes win over non-expansive boxes.
120  Non-empty boxes win over empty boxes.
121 */
122 static inline void qMaxExpCalc(int & max, bool &exp, bool &empty,
123  int boxmax, bool boxexp, bool boxempty)
124 {
125  if (exp) {
126  if (boxexp)
127  max = qMax(max, boxmax);
128  } else {
129  if (boxexp || (empty && (!boxempty || max == 0)))
130  max = boxmax;
131  else if (empty == boxempty)
132  max = qMin(max, boxmax);
133  }
134  exp = exp || boxexp;
135  empty = empty && boxempty;
136 }
