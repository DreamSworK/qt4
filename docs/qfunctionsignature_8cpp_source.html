76 QString FunctionSignature::displayName(const NamePool::Ptr &np) const
77 {
78  QString result;
79  result += np->displayName(name());
80  result += QLatin1Char('(');
81 
82  FunctionArgument::List::const_iterator it(m_arguments.constBegin());
83  const FunctionArgument::List::const_iterator end(m_arguments.constEnd());
84 
85  if(it != end)
86  {
87  while(true)
88  {
89  result += QLatin1Char('$');
90  result += np->displayName((*it)->name());
91  result += QLatin1String(" as ");
92  result += (*it)->type()->displayName(np);
93 
94  ++it;
95  if(it == end)
96  break;
97 
98  result += QLatin1String(", ");
99  }
100  }
101 
102  if(m_maxArgs == FunctionSignature::UnlimitedArity)
103  result += QLatin1String(", ...");
104 
105  result += QLatin1String(") as ");
106  result += m_returnType->displayName(np);
107 
108  return result;
109 }
