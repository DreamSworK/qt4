114 void QProxyModel::setModel(QAbstractItemModel *model)
115 {
116  Q_D(QProxyModel);
117  if (d->model && d->model != &d->empty)
118  disconnectFromModel(d->model);
119  if (model) {
120  d->model = model;
121  connectToModel(model);
122  } else {
123  d->model = &d->empty;
124  }
