254 void QDeclarativeGuardedContextData::setContextData(QDeclarativeContextData *contextData)
255 {
256  clear();
257 
258  if (contextData) {
259  m_contextData = contextData;
260  m_next = contextData->contextGuards;
261  if (m_next) m_next->m_prev = &m_next;
262  m_prev = &contextData->contextGuards;
263  contextData->contextGuards = this;
264  }
265 }

272 void QDeclarativeGuardedContextData::clear()
273 {
274  if (m_prev) {
275  *m_prev = m_next;
276  if (m_next) m_next->m_prev = m_prev;
277  m_contextData = 0;
278  m_next = 0;
279  m_prev = 0;
280  }
281 }
