74 QList<int> QKeyMapper::possibleKeys(QKeyEvent *e)
75 {
76  QList<int> result;
77 
78  if (!e->nativeScanCode()) {
79  if (e->key() && (e->key() != Qt::Key_unknown))
80  result << int(e->key() + e->modifiers());
81  else if (!e->text().isEmpty())

83  return result;
84  }
