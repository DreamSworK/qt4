59 {
60  /* Unfortunately we have to do it this way, QUrl::isParentOf() doesn't
61  * understand URI schemes like this one. */
62  const QString requestedUrl(uri.toString());
63 
64  /* On the topic of timeouts:
65  *
66  * Currently the schemes QNetworkAccessManager handles should/will do
67  * timeouts for 4.4, but we need to do timeouts for our own. */

69  return m_variableURIManager;
70  else
71  {
72  if(!m_genericManager)
