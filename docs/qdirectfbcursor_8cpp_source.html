53 void QDirectFBCursor::changeCursor(QCursor *cursor, QWidget *)
54 {
55  int xSpot;
56  int ySpot;
57  QPixmap map;
58 
59  if (cursor->shape() != Qt::BitmapCursor) {
60  m_image->set(cursor->shape());
61  xSpot = m_image->hotspot().x();
62  ySpot = m_image->hotspot().y();
63  QImage *i = m_image->image();
64  map = QPixmap::fromImage(*i);
65  } else {
66  QPoint point = cursor->hotSpot();
67  xSpot = point.x();
68  ySpot = point.y();
69  map = cursor->pixmap();
70  }
71 
72  DFBResult res;
73  IDirectFBDisplayLayer *layer = toDfbLayer(screen);
74  IDirectFBSurface* surface(QDirectFbConvenience::dfbSurfaceForPlatformPixmap(map.pixmapData()));
75 
76  res = layer->SetCooperativeLevel(layer, DLSCL_ADMINISTRATIVE);
77  if (res != DFB_OK) {
78  DirectFBError("Failed to set DLSCL_ADMINISTRATIVE", res);
79  return;
80  }
81 
82  layer->SetCursorShape(layer, surface, xSpot, ySpot);
83  layer->SetCooperativeLevel(layer, DLSCL_SHARED);
84 }
