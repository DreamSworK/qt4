83 static QDeclarativePrivate::AutoParentResult qgraphicsobject_autoParent(QObject *obj, QObject *parent)
84 {
85  QGraphicsObject* gobj = qobject_cast<QGraphicsObject*>(obj);
86  if (!gobj)
87  return QDeclarativePrivate::IncompatibleObject;
88 
89  QGraphicsObject* gparent = qobject_cast<QGraphicsObject*>(parent);
90  if (!gparent)
91  return QDeclarativePrivate::IncompatibleParent;
92 
93  gobj->setParentItem(gparent);
94  return QDeclarativePrivate::Parented;
95 }
