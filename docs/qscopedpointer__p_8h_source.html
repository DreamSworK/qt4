63 /* Internal helper class - exposes the data through data_ptr (legacy from QShared).
64  Required for some internal Qt classes, do not use otherwise. */
65 template <typename T, typename Cleanup = QScopedPointerDeleter<T> >
66 class QCustomScopedPointer : public QScopedPointer<T, Cleanup>
67 {
68 public:
69  explicit inline QCustomScopedPointer(T *p = 0)
70  : QScopedPointer<T, Cleanup>(p)
71  {
72  }
73

76  return this->d;

93 /* Internal helper class - a handler for QShared* classes, to be used in QCustomScopedPointer */
94 template <typename T>
95 class QScopedPointerSharedDeleter
96 {
97 public:
98  static inline void cleanup(T *d)
99  {
100  if (d && !d->ref.deref())
101  delete d;
102  }
103 };

105 /* Internal.
106  This class is basically a scoped pointer pointing to a ref-counted object
107  */
108 template <typename T>
109 class QScopedSharedPointer : public QCustomScopedPointer<T, QScopedPointerSharedDeleter<T> >
110 {
111 public:
112  explicit inline QScopedSharedPointer(T *p = 0)
113  : QCustomScopedPointer<T, QScopedPointerSharedDeleter<T> >(p)
114  {
115  }
116 
117  inline void detach()
118  {
119  qAtomicDetach(this->d);
120  }
121 
122  inline void assign(T *other)
123  {
124  if (this->d == other)
125  return;
126  if (other)
127  other->ref.ref();
128  T *oldD = this->d;
129  this->d = other;
130  QScopedPointerSharedDeleter<T>::cleanup(oldD);
131  }
132 
133  inline bool operator==(const QScopedSharedPointer<T> &other) const
134  {
135  return this->d == other.d;
136  }
137 
138  inline bool operator!=(const QScopedSharedPointer<T> &other) const
139  {
140  return this->d != other.d;
141  }
142 
143 private:
144  Q_DISABLE_COPY(QScopedSharedPointer)
145 };
