118 
119  switch (hintingPreference) {

121  fe->setDefaultHintStyle(QFontEngineFT::HintNone);
122  break;
123  case QFont::PreferFullHinting:
124  fe->setDefaultHintStyle(QFontEngineFT::HintFull);
125  break;
126  case QFont::PreferVerticalHinting:
127  fe->setDefaultHintStyle(QFontEngineFT::HintLight);
128  break;
129  default:
130  // Leave it as it is
131  break;
132  }
133 
134  fontEngine = fe;
