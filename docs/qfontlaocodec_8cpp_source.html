98 QByteArray QFontLaoCodec::convertFromUnicode(const QChar *uc, int len, ConverterState *) const
99 {
100  QByteArray rstring(len, Qt::Uninitialized);
101  uchar *rdata = (uchar *) rstring.data();
102  const QChar *sdata = uc;
103  int i = 0;
104  for (; i < len; ++i, ++sdata, ++rdata) {
105  if (sdata->unicode() < 0x80) {
106  *rdata = (uchar) sdata->unicode();
107  } else if (sdata->unicode() >= 0x0e80 && sdata->unicode() <= 0x0eff) {
108  uchar lao = unicode_to_mulelao[sdata->unicode() - 0x0e80];
109  if (lao)
110  *rdata = lao;
111  else
112  *rdata = 0;
113  } else {
114  *rdata = 0;
115  }
116  }
117  return rstring;
118 }
