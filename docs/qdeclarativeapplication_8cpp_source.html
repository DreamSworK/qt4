85 bool QDeclarativeApplication::eventFilter(QObject *obj, QEvent *event)
86 {
87  Q_UNUSED(obj)
88  Q_D(QDeclarativeApplication);
89  if (event->type() == QEvent::ApplicationActivate
90  || event->type() == QEvent::ApplicationDeactivate) {
91  bool active = d->active;
92  if (event->type() == QEvent::ApplicationActivate)
93  active = true;
94  else if (event->type() == QEvent::ApplicationDeactivate)
95  active = false;
96 
97  if (d->active != active) {
98  d->active = active;
99  emit activeChanged();
100  }
101  }
102  if (event->type() == QEvent::LayoutDirectionChange) {
103  Qt::LayoutDirection direction = QApplication::layoutDirection();
104  if (d->layoutDirection != direction) {
105  d->layoutDirection = direction;
106  emit layoutDirectionChanged();
107  }
108  }
109  return false;
110 }
