82  inline void transformSystemClip()
83  {
84  if (systemClip.isEmpty())
85  return;
86 
87  if (hasSystemTransform) {
88  if (systemTransform.type() <= QTransform::TxTranslate)
89  systemClip.translate(qRound(systemTransform.dx()), qRound(systemTransform.dy()));
90  else
91  systemClip = systemTransform.map(systemClip);
92  }
93 
94  // Make sure we're inside the viewport.
95  if (hasSystemViewport) {
96  systemClip &= systemViewport;
97  if (systemClip.isEmpty()) {
98  // We don't want to paint without system clip, so set it to 1 pixel :)
99  systemClip = QRect(systemViewport.boundingRect().topLeft(), QSize(1, 1));
100  }
101  }
102  }
103 
104  inline void setSystemTransform(const QTransform &xform)
105  {
106  systemTransform = xform;
107  if ((hasSystemTransform = !xform.isIdentity()) || hasSystemViewport)
108  transformSystemClip();

110  }
