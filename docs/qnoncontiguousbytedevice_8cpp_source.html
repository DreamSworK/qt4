334 const char* QNonContiguousByteDeviceIoDeviceImpl::readPointer(qint64 maximumLength, qint64 &len)
335 {
336  if (eof == true) {
337  len = -1;
338  return 0;
339  }
340 
341  if (currentReadBuffer == 0)
342  currentReadBuffer = new QByteArray(currentReadBufferSize, '\0'); // lazy alloc
343 
344  if (maximumLength == -1)
345  maximumLength = currentReadBufferSize;
346 
347  if (currentReadBufferAmount - currentReadBufferPosition > 0) {
348  len = currentReadBufferAmount - currentReadBufferPosition;
349  return currentReadBuffer->data() + currentReadBufferPosition;
350  }
351 
352  qint64 haveRead = device->read(currentReadBuffer->data(), qMin(maximumLength, currentReadBufferSize));
353 
354  if ((haveRead == -1) || (haveRead == 0 && device->atEnd() && !device->isSequential())) {
355  eof = true;
356  len = -1;
357  // size was unknown before, emit a readProgress with the final size
358  if (size() == -1)
359  emit readProgress(totalAdvancements, totalAdvancements);
360  return 0;
361  }
362 
363  currentReadBufferAmount = haveRead;
364  currentReadBufferPosition = 0;
365 
366  len = haveRead;
367  return currentReadBuffer->data();
368 }

370 bool QNonContiguousByteDeviceIoDeviceImpl::advanceReadPointer(qint64 amount)
371 {
372  totalAdvancements += amount;
373 
374  // normal advancement
375  currentReadBufferPosition += amount;
376 
377  if (size() == -1)
378  emit readProgress(totalAdvancements, totalAdvancements);
379  else
380  emit readProgress(totalAdvancements, size());
381 
382  // advancing over that what has actually been read before
383  if (currentReadBufferPosition > currentReadBufferAmount) {
384  qint64 i = currentReadBufferPosition - currentReadBufferAmount;
385  while (i > 0) {
386  if (device->getChar(0) == false) {
387  emit readProgress(totalAdvancements - i, size());
388  return false; // ### FIXME handle eof
389  }
390  i--;
391  }
392 
393  currentReadBufferPosition = 0;
394  currentReadBufferAmount = 0;
395  }
396 
397 
398  return true;
399 }

406 bool QNonContiguousByteDeviceIoDeviceImpl::reset()
407 {
408  if (resetDisabled)
409  return false;
410  bool reset = (initialPosition == 0) ? device->reset() : device->seek(initialPosition);
411  if (reset) {
412  eof = false; // assume eof is false, it will be true after a read has been attempted
413  totalAdvancements = 0; //reset the progress counter
414  if (currentReadBuffer) {
415  delete currentReadBuffer;
416  currentReadBuffer = 0;
417  }
418  currentReadBufferAmount = 0;
419  currentReadBufferPosition = 0;
420  return true;
421  }
422 
423  return false;
424 }
