293 void QDeclarativeTimer::update()
294 {
295  Q_D(QDeclarativeTimer);
296  if (d->classBegun && !d->componentComplete)
297  return;
298  d->pause.stop();
299  if (d->running) {
300  d->pause.setCurrentTime(0);
301  d->pause.setLoopCount(d->repeating ? -1 : 1);
302  d->pause.setDuration(d->interval);
303  d->pause.start();
304  if (d->triggeredOnStart && d->firstTick) {
305  QCoreApplication::removePostedEvents(this, QEvent::MetaCall);
306  QMetaObject::invokeMethod(this, "ticked", Qt::QueuedConnection);
307  }
308  }
309 }

332 void QDeclarativeTimer::ticked()
333 {
334  Q_D(QDeclarativeTimer);
335  if (d->running && (d->pause.currentTime() > 0 || (d->triggeredOnStart && d->firstTick)))
336  emit triggered();
337  d->firstTick = false;
338 }

340 void QDeclarativeTimer::finished()
341 {
342  Q_D(QDeclarativeTimer);
343  if (d->repeating || !d->running)
344  return;
345  emit triggered();
346  d->running = false;
347  d->firstTick = false;
348  emit runningChanged();
349 }
