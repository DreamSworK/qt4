67 QList<int> QKeyMapperPrivate::possibleKeys(QKeyEvent *e)
68 {
69  QList<int> result;
70  if (e->key() && (e->key() != Qt::Key_unknown))
71  result << int(e->key() + e->modifiers());
72  else if (!e->text().isEmpty())
73  result << int(e->text().at(0).unicode() + e->modifiers());
74  return result;
