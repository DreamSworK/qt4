309 QList<QNetworkProxy> QGlobalNetworkProxy::proxyForQuery(const QNetworkProxyQuery &query)
310 {
311  QMutexLocker locker(&mutex);
312 
313  QList<QNetworkProxy> result;
314 
315  // don't look for proxies for a local connection
316  QHostAddress parsed;
317  QString hostname = query.url().host();
318  if (hostname == QLatin1String("localhost")
319  || hostname.startsWith(QLatin1String("localhost."))
320  || (parsed.setAddress(hostname)
321  && (parsed == QHostAddress::LocalHost
322  || parsed == QHostAddress::LocalHostIPv6))) {
323  result << QNetworkProxy(QNetworkProxy::NoProxy);
324  return result;
325  }
326 
327  if (!applicationLevelProxyFactory) {
328  if (applicationLevelProxy
329  && applicationLevelProxy->type() != QNetworkProxy::DefaultProxy)
330  result << *applicationLevelProxy;
331  else
332  result << QNetworkProxy(QNetworkProxy::NoProxy);
333  return result;
334  }
335 
336  // we have a factory
337  result = applicationLevelProxyFactory->queryProxy(query);
338  if (result.isEmpty()) {
339  qWarning("QNetworkProxyFactory: factory %p has returned an empty result set",
340  applicationLevelProxyFactory);
341  result << QNetworkProxy(QNetworkProxy::NoProxy);
342  }
343  return result;
344 }

356 static QNetworkProxy::Capabilities defaultCapabilitiesForType(QNetworkProxy::ProxyType type)
357 {
358  q_static_assert(int(QNetworkProxy::DefaultProxy) == 0);
359  q_static_assert(int(QNetworkProxy::FtpCachingProxy) == 5);
360  static const int defaults[] =
361  {
362  /* [QNetworkProxy::DefaultProxy] = */
363  (int(QNetworkProxy::ListeningCapability) |
364  int(QNetworkProxy::TunnelingCapability) |
365  int(QNetworkProxy::UdpTunnelingCapability)),
366  /* [QNetworkProxy::Socks5Proxy] = */
367  (int(QNetworkProxy::TunnelingCapability) |
368  int(QNetworkProxy::ListeningCapability) |
369  int(QNetworkProxy::UdpTunnelingCapability) |
370  int(QNetworkProxy::HostNameLookupCapability)),
371  // it's weird to talk about the proxy capabilities of a "not proxy"...
372  /* [QNetworkProxy::NoProxy] = */
373  (int(QNetworkProxy::ListeningCapability) |
374  int(QNetworkProxy::TunnelingCapability) |
375  int(QNetworkProxy::UdpTunnelingCapability)),
376  /* [QNetworkProxy::HttpProxy] = */
377  (int(QNetworkProxy::TunnelingCapability) |
378  int(QNetworkProxy::CachingCapability) |
379  int(QNetworkProxy::HostNameLookupCapability)),
380  /* [QNetworkProxy::HttpCachingProxy] = */
381  (int(QNetworkProxy::CachingCapability) |
382  int(QNetworkProxy::HostNameLookupCapability)),
383  /* [QNetworkProxy::FtpCachingProxy] = */
384  (int(QNetworkProxy::CachingCapability) |
385  int(QNetworkProxy::HostNameLookupCapability)),
386  };
387 
388  if (int(type) < 0 || int(type) > int(QNetworkProxy::FtpCachingProxy))
389  type = QNetworkProxy::DefaultProxy;
390  return QNetworkProxy::Capabilities(defaults[int(type)]);
391 }
