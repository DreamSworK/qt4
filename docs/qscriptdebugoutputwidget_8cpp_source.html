119 void QScriptDebugOutputWidget::message(
120  QtMsgType type, const QString &text, const QString &fileName,
121  int lineNumber, int columnNumber, const QVariant &/*data*/)
122 {
123  // ### unify with QScriptDebuggerConsoleWidget::message()
124  Q_D(QScriptDebugOutputWidget);
125  QString msg;
126  if (!fileName.isEmpty() || (lineNumber != -1)) {
127  if (!fileName.isEmpty())
128  msg.append(fileName);
129  else
130  msg.append(QLatin1String("<noname>"));
131  if (lineNumber != -1) {
132  msg.append(QLatin1Char(':'));
133  msg.append(QString::number(lineNumber));
134  if (columnNumber != -1) {
135  msg.append(QLatin1Char(':'));
136  msg.append(QString::number(columnNumber));
137  }
138  }
139  msg.append(QLatin1String(": "));
140  }
141  msg.append(text);
142  QTextCharFormat oldFmt = d->outputEdit->currentCharFormat();
143  QTextCharFormat fmt(oldFmt);
144  if (type == QtCriticalMsg) {
145  fmt.setForeground(Qt::red);
146  d->outputEdit->setCurrentCharFormat(fmt);
147  }
148  d->outputEdit->appendPlainText(msg);
149  d->outputEdit->setCurrentCharFormat(oldFmt);
150  d->outputEdit->scrollToBottom();
151 }
