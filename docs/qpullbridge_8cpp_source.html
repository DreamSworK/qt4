73 AbstractXmlPullProvider::Event PullBridge::next()
74 {
75  m_index = m_iterators.top().second->next();
76 
77  if(!m_index.isNull())
78  {
79  Item item(m_index);
80 
81  if(item && item.isAtomicValue())
82  m_current = AtomicValue;
83  else
84  {
85  Q_ASSERT(item.isNode());
86 
87  switch(m_index.kind())
88  {
89  case QXmlNodeModelIndex::Attribute:
90  {
91  m_current = Attribute;
92  break;
93  }
94  case QXmlNodeModelIndex::Comment:
95  {
96  m_current = Comment;
97  break;
98  }
99  case QXmlNodeModelIndex::Element:
100  {
101  m_iterators.push(qMakePair(StartElement, m_index.iterate(QXmlNodeModelIndex::AxisChild)));
102  m_current = StartElement;
103  break;
104  }
105  case QXmlNodeModelIndex::Document:
106  {
107  m_iterators.push(qMakePair(StartDocument, m_index.iterate(QXmlNodeModelIndex::AxisChild)));
108  m_current = StartDocument;
109  break;
110  }
111  case QXmlNodeModelIndex::Namespace:
112  {
113  m_current = Namespace;
114  break;
115  }
116  case QXmlNodeModelIndex::ProcessingInstruction:
117  {
118  m_current = ProcessingInstruction;
119  break;
120  }
121  case QXmlNodeModelIndex::Text:
122  {
123  m_current = Text;
124  break;
125  }
126  }
127  }
128  }
129  else
130  {
131  if(m_iterators.isEmpty())
132  m_current = EndOfInput;
133  else
134  {
135  switch(m_iterators.top().first)
136  {
137  case StartOfInput:
138  {
139  m_current = EndOfInput;
140  break;
141  }
142  case StartElement:
143  {
144  m_current = EndElement;
145  m_iterators.pop();
146  break;
147  }
148  case StartDocument:
149  {
150  m_current = EndDocument;
151  m_iterators.pop();
152  break;
153  }
154  default:
155  {
156  Q_ASSERT_X(false, Q_FUNC_INFO,
157  "Invalid value.");
158  m_current = EndOfInput;
159  }
160  }
161  }
162 
163  }
164 
165  return m_current;
166 }
