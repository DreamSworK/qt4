67 class QML_PARSER_EXPORT MemoryPool : public QSharedData
68 {
69 public:
70  enum { maxBlockCount = -1 };
71  enum { defaultBlockSize = 1 << 12 };
72 
73  MemoryPool() {
74  m_blockIndex = maxBlockCount;
75  m_currentIndex = 0;
76  m_storage = 0;
77  m_currentBlock = 0;
78  m_currentBlockSize = 0;
79  }
80 
81  virtual ~MemoryPool() {
82  for (int index = 0; index < m_blockIndex + 1; ++index)
83  qFree(m_storage[index]);
84 
85  qFree(m_storage);
86  }
87 
88  char *allocate(int bytes) {
89  bytes += (8 - bytes) & 7; // ensure multiple of 8 bytes (maintain alignment)
90  if (m_currentBlock == 0 || m_currentBlockSize < m_currentIndex + bytes) {
91  ++m_blockIndex;
92  m_currentBlockSize = defaultBlockSize << m_blockIndex;
93 
94  m_storage = reinterpret_cast<char**>(qRealloc(m_storage, sizeof(char*) * (1 + m_blockIndex)));
95  m_currentBlock = m_storage[m_blockIndex] = reinterpret_cast<char*>(qMalloc(m_currentBlockSize));
96  ::memset(m_currentBlock, 0, m_currentBlockSize);
97 
98  m_currentIndex = (8 - quintptr(m_currentBlock)) & 7; // ensure first chunk is 64-bit aligned
99  Q_ASSERT(m_currentIndex + bytes <= m_currentBlockSize);
100  }
101 
102  char *p = reinterpret_cast<char *>
103  (m_currentBlock + m_currentIndex);
104 
105  m_currentIndex += bytes;
106 
107  return p;
108  }
109 
110  int bytesAllocated() const {
111  int bytes = 0;
112  for (int index = 0; index < m_blockIndex; ++index)
113  bytes += (defaultBlockSize << index);
114  bytes += m_currentIndex;
115  return bytes;
116  }
117 
118 private:
119  int m_blockIndex;
120  int m_currentIndex;
121  char *m_currentBlock;
122  int m_currentBlockSize;
123  char **m_storage;
124 
125 private:
126  Q_DISABLE_COPY(MemoryPool)
127 };
