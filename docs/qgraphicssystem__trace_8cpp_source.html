79 QTraceWindowSurface::~QTraceWindowSurface()
80 {
81  if (buffer) {
82  QFile outputFile(QString(QLatin1String("qtgraphics-%0.trace")).arg(winId));
83  if (outputFile.open(QIODevice::WriteOnly)) {
84  QDataStream out(&outputFile);
85  out.setFloatingPointPrecision(QDataStream::SinglePrecision);
86 
87  out.writeBytes("qttraceV2", 9);
88 
89  uint version = 1;
90 
91  out << version << *buffer << updates;
92  }
93  delete buffer;
94  }
95 }

97 QPaintDevice *QTraceWindowSurface::paintDevice()

106  }
107  return buffer;
108 }

119 void QTraceWindowSurface::endPaint(const QRegion &rgn)
120 {
121  QPainter p(QRasterWindowSurface::paintDevice());
122  buffer->draw(&p, buffer->numFrames()-1);
123  p.end();
124

126 
127  updates << rgn;
