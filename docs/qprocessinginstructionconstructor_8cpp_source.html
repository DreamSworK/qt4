59 QString ProcessingInstructionConstructor::leftTrimmed(const QString &input)
60 {

62 
63  for(int i = 0; i < len; ++i)
64  {
65  if(!input.at(i).isSpace())

68 
69  return QString(); /* input consists only of whitespace. All was trimmed. */
