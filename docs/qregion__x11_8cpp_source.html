50 void QRegion::updateX11Region() const
51 {
52  d->rgn = XCreateRegion();
53  if (!d->qt_rgn)
54  return;
55 
56  int n = d->qt_rgn->numRects;
57  const QRect *rect = (n == 1 ? &d->qt_rgn->extents : d->qt_rgn->rects.constData());
58  while (n--) {
59  XRectangle r;
60  r.x = qMax(SHRT_MIN, rect->x());
61  r.y = qMax(SHRT_MIN, rect->y());
62  r.width = qMin((int)USHRT_MAX, rect->width());
63  r.height = qMin((int)USHRT_MAX, rect->height());
64  XUnionRectWithRegion(&r, d->rgn, d->rgn);
65  ++rect;
66  }
67 }

69 void *QRegion::clipRectangles(int &num) const
70 {
71  if (!d->xrectangles && !(d == &shared_empty || d->qt_rgn->numRects == 0)) {
72  XRectangle *r = static_cast<XRectangle*>(malloc(d->qt_rgn->numRects * sizeof(XRectangle)));
73  d->xrectangles = r;
74  int n = d->qt_rgn->numRects;
75  const QRect *rect = (n == 1 ? &d->qt_rgn->extents : d->qt_rgn->rects.constData());
76  while (n--) {
77  r->x = qMax(SHRT_MIN, rect->x());
78  r->y = qMax(SHRT_MIN, rect->y());
79  r->width = qMin((int)USHRT_MAX, rect->width());
80  r->height = qMin((int)USHRT_MAX, rect->height());
81  ++r;
82  ++rect;
83  }
84  }
85  if (d == &shared_empty || d->qt_rgn->numRects == 0)
86  num = 0;
87  else
88  num = d->qt_rgn->numRects;
89  return d->xrectangles;
90 }
