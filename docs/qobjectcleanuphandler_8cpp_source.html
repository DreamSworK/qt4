98 QObject *QObjectCleanupHandler::add(QObject* object)
99 {

101  return 0;
102 
103  connect(object, SIGNAL(destroyed(QObject*)), this, SLOT(objectDestroyed(QObject*)));
104  cleanupObjects.insert(0, object);

106 }

114 void QObjectCleanupHandler::remove(QObject *object)
