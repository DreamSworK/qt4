207 void QDeclarativeEngineDebugPrivate::decode(QDataStream &ds, QDeclarativeDebugObjectReference &o,
208  bool simple)
209 {
210  QDeclarativeEngineDebugService::QDeclarativeObjectData data;
211  ds >> data;
212  o.m_debugId = data.objectId;
213  o.m_class = data.objectType;
214  o.m_idString = data.idString;
215  o.m_name = data.objectName;
216  o.m_source.m_url = data.url;
217  o.m_source.m_lineNumber = data.lineNumber;
218  o.m_source.m_columnNumber = data.columnNumber;
219  o.m_contextDebugId = data.contextId;
220  o.m_parentId = data.parentId;
221  o.m_needsMoreData = simple;
222 
223  if (simple)
224  return;
225 
226  int childCount;
227  bool recur;
228  ds >> childCount >> recur;
229 
230  for (int ii = 0; ii < childCount; ++ii) {
231  o.m_children.append(QDeclarativeDebugObjectReference());
232  decode(ds, o.m_children.last(), !recur);
233  }
234 
235  int propCount;
236  ds >> propCount;
237 
238  for (int ii = 0; ii < propCount; ++ii) {
239  QDeclarativeEngineDebugService::QDeclarativeObjectProperty data;
240  ds >> data;
241  QDeclarativeDebugPropertyReference prop;
242  prop.m_objectDebugId = o.m_debugId;
243  prop.m_name = data.name;
244  prop.m_binding = data.binding;
245  prop.m_hasNotifySignal = data.hasNotifySignal;
246  prop.m_valueTypeName = data.valueTypeName;
247  switch (data.type) {
248  case QDeclarativeEngineDebugService::QDeclarativeObjectProperty::Basic:
249  case QDeclarativeEngineDebugService::QDeclarativeObjectProperty::List:
250  case QDeclarativeEngineDebugService::QDeclarativeObjectProperty::SignalProperty:
251  case QDeclarativeEngineDebugService::QDeclarativeObjectProperty::Variant:
252  {
253  prop.m_value = data.value;
254  break;
255  }
256  case QDeclarativeEngineDebugService::QDeclarativeObjectProperty::Object:
257  {
258  QDeclarativeDebugObjectReference obj;
259  obj.m_debugId = prop.m_value.toInt();
260  prop.m_value = QVariant::fromValue(obj);
261  break;
262  }
263  case QDeclarativeEngineDebugService::QDeclarativeObjectProperty::Unknown:
264  break;
265  }
266  o.m_properties << prop;
267  }
268 }

299 void QDeclarativeEngineDebugPrivate::message(const QByteArray &data)
300 {
301  QDataStream ds(data);
302 
303  QByteArray type;
304  ds >> type;
305 
306  //qDebug() << "QDeclarativeEngineDebugPrivate::message()" << type;
307 
308  if (type == "LIST_ENGINES_R") {
309  int queryId;
310  ds >> queryId;
311 
312  QDeclarativeDebugEnginesQuery *query = enginesQuery.value(queryId);
313  if (!query)
314  return;
315  enginesQuery.remove(queryId);
316 
317  int count;
318  ds >> count;
319 
320  for (int ii = 0; ii < count; ++ii) {
321  QDeclarativeDebugEngineReference ref;
322  ds >> ref.m_name;
323  ds >> ref.m_debugId;
324  query->m_engines << ref;
325  }
326 
327  query->m_client = 0;
328  query->setState(QDeclarativeDebugQuery::Completed);
329  } else if (type == "LIST_OBJECTS_R") {
330  int queryId;
331  ds >> queryId;
332 
333  QDeclarativeDebugRootContextQuery *query = rootContextQuery.value(queryId);
334  if (!query)
335  return;
336  rootContextQuery.remove(queryId);
337 
338  if (!ds.atEnd())
339  decode(ds, query->m_context);
340 
341  query->m_client = 0;
342  query->setState(QDeclarativeDebugQuery::Completed);
343  } else if (type == "FETCH_OBJECT_R") {
344  int queryId;
345  ds >> queryId;
346 
347  QDeclarativeDebugObjectQuery *query = objectQuery.value(queryId);
348  if (!query)
349  return;
350  objectQuery.remove(queryId);
351 
352  if (!ds.atEnd())
353  decode(ds, query->m_object, false);
354 
355  query->m_client = 0;
356  query->setState(QDeclarativeDebugQuery::Completed);
357  } else if (type == "EVAL_EXPRESSION_R") {
358  int queryId;
359  QVariant result;
360  ds >> queryId >> result;
361 
362  QDeclarativeDebugExpressionQuery *query = expressionQuery.value(queryId);
363  if (!query)
364  return;
365  expressionQuery.remove(queryId);
366 
367  query->m_result = result;
368  query->m_client = 0;
369  query->setState(QDeclarativeDebugQuery::Completed);
370  } else if (type == "WATCH_PROPERTY_R") {
371  int queryId;
372  bool ok;
373  ds >> queryId >> ok;
374 
375  QDeclarativeDebugWatch *watch = watched.value(queryId);
376  if (!watch)
377  return;
378 
379  watch->setState(ok ? QDeclarativeDebugWatch::Active : QDeclarativeDebugWatch::Inactive);
380  } else if (type == "WATCH_OBJECT_R") {
381  int queryId;
382  bool ok;
383  ds >> queryId >> ok;
384 
385  QDeclarativeDebugWatch *watch = watched.value(queryId);
386  if (!watch)
387  return;
388 
389  watch->setState(ok ? QDeclarativeDebugWatch::Active : QDeclarativeDebugWatch::Inactive);
390  } else if (type == "WATCH_EXPR_OBJECT_R") {
391  int queryId;
392  bool ok;
393  ds >> queryId >> ok;
394 
395  QDeclarativeDebugWatch *watch = watched.value(queryId);
396  if (!watch)
397  return;
398 
399  watch->setState(ok ? QDeclarativeDebugWatch::Active : QDeclarativeDebugWatch::Inactive);
400  } else if (type == "UPDATE_WATCH") {
401  int queryId;
402  int debugId;
403  QByteArray name;
404  QVariant value;
405  ds >> queryId >> debugId >> name >> value;
406 
407  QDeclarativeDebugWatch *watch = watched.value(queryId, 0);
408  if (!watch)
409  return;
410  emit watch->valueChanged(name, value);
411  } else if (type == "OBJECT_CREATED") {
412  emit q_func()->newObjects();
413  }
414 }
