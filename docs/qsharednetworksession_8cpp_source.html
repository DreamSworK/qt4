67 QSharedPointer<QNetworkSession> QSharedNetworkSessionManager::getSession(QNetworkConfiguration config)
68 {
69  QSharedNetworkSessionManager *m(sharedNetworkSessionManager());
70  //if already have a session, return it
71  if (m->sessions.contains(config)) {
72  QSharedPointer<QNetworkSession> p = m->sessions.value(config).toStrongRef();
