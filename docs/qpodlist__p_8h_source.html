64 template <typename T, int Prealloc>
65 class QPodList : public QVarLengthArray<T, Prealloc>
66 {

72  inline explicit QPodList(int size = 0)
73  : QVarLengthArray<T, Prealloc>(size)
74  {}

76  inline void insert(int idx, const T &t)
77  {
78  const int sz = s++;
79  if (s == a)
80  realloc(s, s << 1);
81  ::memmove(ptr + idx + 1, ptr + idx, (sz - idx) * sizeof(T));
82  ptr[idx] = t;
83  }
84 
85  inline void removeAll(const T &t)
86  {
87  int i = 0;
88  for (int j = 0; j < s; ++j) {
89  if (ptr[j] != t)
90  ptr[i++] = ptr[j];
91  }
92  s = i;
93  }
94 
95  inline void removeAt(int idx)
96  {
97  Q_ASSERT(idx >= 0 && idx < s);
98  ::memmove(ptr + idx, ptr + idx + 1, (s - idx - 1) * sizeof(T));
99  --s;
100  }
101 
102  inline T takeFirst()
103  {
104  Q_ASSERT(s > 0);
105  T tmp = ptr[0];
106  removeAt(0);
107  return tmp;
108  }
109 };
