59 ConstructorFunctionsFactory::ConstructorFunctionsFactory(const NamePool::Ptr &np, const SchemaTypeFactory::Ptr &f) : m_typeFactory(f)
60 {
61  Q_ASSERT(m_typeFactory);
62  Q_ASSERT(np);
63  SchemaType::Hash::const_iterator it(m_typeFactory->types().constBegin());
64  const SchemaType::Hash::const_iterator end(m_typeFactory->types().constEnd());
65 
66  FunctionArgument::List args;
67  const QXmlName argName(StandardNamespaces::empty, StandardLocalNames::sourceValue);
68 
69  args.append(FunctionArgument::Ptr(new FunctionArgument(argName,
70  CommonSequenceTypes::ZeroOrOneAtomicType)));
71 
72  while(it != end)
73  {
74  if(!BuiltinTypes::xsAnyAtomicType->wxsTypeMatches(*it) ||
75  *BuiltinTypes::xsAnyAtomicType == *static_cast<const AtomicType *>((*it).data()) ||
76  *BuiltinTypes::xsNOTATION == *static_cast<const AtomicType *>((*it).data()))
77  {
78  /* It's not a valid type for a constructor function -- skip it. */
79  ++it;
80  continue;
81  }
82 
83  const QXmlName name((*it)->name(np));
84  FunctionSignature::Ptr s(new FunctionSignature(name, 1, 1,
85  makeGenericSequenceType(AtomicType::Ptr(*it),
86  Cardinality::zeroOrOne())));
87  s->setArguments(args);
88  m_signatures.insert(name, s);
89  ++it;
90  }
91 }
