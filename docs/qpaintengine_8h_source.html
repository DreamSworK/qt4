332 inline void QPaintEngine::fix_neg_rect(int *x, int *y, int *w, int *h)
333 {
334  if (*w < 0) {
335  *w = -*w;
336  *x -= *w - 1;
337  }
338  if (*h < 0) {
339  *h = -*h;
340  *y -= *h - 1;
341  }
