55 QList<QPrinterInfo> QPrinterInfo::availablePrinters()
56 {
57  QList<QPrinterInfo> printers;
58 
59  DWORD needed = 0;
60  DWORD returned = 0;
61  if (!EnumPrinters(PRINTER_ENUM_LOCAL | PRINTER_ENUM_CONNECTIONS, NULL, 4, 0, 0, &needed, &returned)) {
62  LPBYTE buffer = new BYTE[needed];
63  if (EnumPrinters(PRINTER_ENUM_LOCAL | PRINTER_ENUM_CONNECTIONS, NULL, 4, buffer, needed, &needed, &returned)) {
64  PPRINTER_INFO_4 infoList = reinterpret_cast<PPRINTER_INFO_4>(buffer);
65  QPrinterInfo defPrn = defaultPrinter();
66  for (uint i = 0; i < returned; ++i) {
67  QString printerName(QString::fromWCharArray(infoList[i].pPrinterName));
68 
69  QPrinterInfo printerInfo(printerName);
70  if (printerInfo.printerName() == defPrn.printerName())
71  printerInfo.d_ptr->isDefault = true;
72  printers.append(printerInfo);
73  }
74  }
75  delete [] buffer;
76  }
77 
78  return printers;
79 }

81 QPrinterInfo QPrinterInfo::defaultPrinter()
82 {
83  QString noPrinters(QLatin1String("qt_no_printers"));
84  wchar_t buffer[256];
85  GetProfileString(L"windows", L"device", (wchar_t*)noPrinters.utf16(), buffer, 256);
86  QString output = QString::fromWCharArray(buffer);
87  if (output != noPrinters) {
88  // Filter out the name of the printer, which should be everything before a comma.
89  QString printerName = output.split(QLatin1Char(',')).value(0);
90  QPrinterInfo printerInfo(printerName);
91  printerInfo.d_ptr->isDefault = true;
92  return printerInfo;
93  }
94 
95  return QPrinterInfo();
96 }
