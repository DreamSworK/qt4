116 inline int qt_depthForFormat(QImage::Format format)
117 {
118  int depth = 0;
119  switch(format) {
120  case QImage::Format_Invalid:
121  case QImage::NImageFormats:
122  Q_ASSERT(false);
123  case QImage::Format_Mono:
124  case QImage::Format_MonoLSB:
125  depth = 1;
126  break;
127  case QImage::Format_Indexed8:
128  depth = 8;
129  break;
130  case QImage::Format_RGB32:
131  case QImage::Format_ARGB32:
132  case QImage::Format_ARGB32_Premultiplied:
133  depth = 32;
134  break;
135  case QImage::Format_RGB555:
136  case QImage::Format_RGB16:
137  case QImage::Format_RGB444:
138  case QImage::Format_ARGB4444_Premultiplied:
139  depth = 16;
140  break;
141  case QImage::Format_RGB666:
142  case QImage::Format_ARGB6666_Premultiplied:
143  case QImage::Format_ARGB8565_Premultiplied:
144  case QImage::Format_ARGB8555_Premultiplied:
145  case QImage::Format_RGB888:
146  depth = 24;
147  break;
148  }
149  return depth;
150 }
