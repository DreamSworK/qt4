765 int QtPrivate::QStringList_removeDuplicates(QStringList *that)
766 {
767  int n = that->size();
768  int j = 0;
769  QSet<QString> seen;
770  seen.reserve(n);
771  for (int i = 0; i < n; ++i) {
772  const QString &s = that->at(i);
773  if (seen.contains(s))
774  continue;
775  seen.insert(s);
776  if (j != i)
777  (*that)[j] = s;
778  ++j;
779  }
780  if (n != j)
781  that->erase(that->begin() + j, that->end());
782  return n - j;
783 }
