268 void QInputContext::sendEvent(const QInputMethodEvent &event)
269 {
270  // route events over input context parents to make chaining possible.
271  QInputContext *p = qobject_cast<QInputContext *>(parent());
272  if (p) {
273  p->sendEvent(event);
274  return;
275  }
276 
277  QWidget *focus = focusWidget();
278  if (!focus)
279  return;
280 
281  QInputMethodEvent e(event);
282  QApplication::sendEvent(focus, &e);
283 }

435 QTextFormat QInputContext::standardFormat(StandardFormat s) const
436 {
437  QWidget *focus = focusWidget();
438  const QPalette &pal = focus ? focus->palette() : QApplication::palette();
439 
440  QTextCharFormat fmt;
441  QColor bg;
442  switch (s) {
443  case QInputContext::PreeditFormat: {
444  fmt.setUnderlineStyle(QTextCharFormat::DashUnderline);
445  break;
446  }
447  case QInputContext::SelectionFormat: {
448  bg = pal.text().color();
449  fmt.setBackground(QBrush(bg));
450  fmt.setForeground(pal.background());
451  break;
452  }
453  }
454  return fmt;
455 }
