228 inline qreal QTransform::determinant() const
229 {
230  return affine._m11*(m_33*affine._m22-affine._dy*m_23) -
231  affine._m21*(m_33*affine._m12-affine._dy*m_13)+affine._dx*(m_23*affine._m12-affine._m22*m_13);
232 }

282 inline QTransform &QTransform::operator*=(qreal num)
283 {
284  if (num == 1.)
285  return *this;
286  affine._m11 *= num;
287  affine._m12 *= num;
288  m_13 *= num;
289  affine._m21 *= num;
290  affine._m22 *= num;
291  m_23 *= num;
292  affine._dx *= num;
293  affine._dy *= num;
294  m_33 *= num;
295  if (m_dirty < TxScale)
296  m_dirty = TxScale;
297  return *this;
298 }
