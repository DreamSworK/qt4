338 bool QDeclarativeSystemPalette::eventFilter(QObject *watched, QEvent *event)
339 {
340  if (watched == qApp) {
341  if (event->type() == QEvent::ApplicationPaletteChange) {
342  QApplication::postEvent(this, new QEvent(QEvent::ApplicationPaletteChange));
343  return false;
344  }
345  }
346  return QObject::eventFilter(watched, event);
347 }

349 bool QDeclarativeSystemPalette::event(QEvent *event)
350 {
351  Q_D(QDeclarativeSystemPalette);
352  if (event->type() == QEvent::ApplicationPaletteChange) {
353  d->palette = QApplication::palette();
354  emit paletteChanged();
355  return true;
356  }
357  return QObject::event(event);
358 }
